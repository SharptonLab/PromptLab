# Probiotic intervention microbiome ordination + PERMANOVA
#
# Input:
#   data/probiotic_study.rds

# Outputs:
#   results/pcoa_bray_curtis_treatment_timepoint.pdf
#   PERMANOVA summary printed to console
#
# Assumptions:
#   - Input RDS contains a valid phyloseq object.

#       sample_id, subject_id, timepoint, treatment, age, sex

#   - subject_id identifies repeated measures.

############################################################
# 1. Setup
############################################################

# ---- Package checks ----
required_packages <- c("phyloseq", "vegan", "ggplot2")

for (pkg in required_packages) {
  if (!requireNamespace(pkg, quietly = TRUE)) {
    stop(
      sprintf(
        "Package '%s' is required but not installed. Please install it before running this script.",
        pkg
      )
    )
  }
}

suppressPackageStartupMessages({
  library(phyloseq)
  library(vegan)
  library(ggplot2)
})

# ---- File paths ----
input_rds_path <- "data/probiotic_study.rds"
output_plot_path <- "results/pcoa_bray_curtis_treatment_timepoint.pdf"

# Create output directory if needed
dir.create(dirname(output_plot_path), recursive = TRUE, showWarnings = FALSE)

# ---- Analysis settings ----
set.seed(123)

expected_timepoint_levels <- c("baseline", "week4", "week8")
expected_treatment_levels <- c("probiotic", "placebo")
permanova_permutations <- 9999


############################################################
# 2. Load phyloseq object and perform basic format checks
############################################################

if (!file.exists(input_rds_path)) {
  stop(sprintf("Input file not found: %s", input_rds_path))
}

physeq <- readRDS(input_rds_path)

if (!inherits(physeq, "phyloseq")) {
  stop("The loaded RDS object is not a phyloseq object.")
}

if (is.null(otu_table(physeq, errorIfNULL = FALSE))) {
  stop("The phyloseq object does not contain an OTU/ASV table.")
}

if (is.null(tax_table(physeq, errorIfNULL = FALSE))) {
  stop("The phyloseq object does not contain a taxonomy table.")
}

if (is.null(sample_data(physeq, errorIfNULL = FALSE))) {
  stop("The phyloseq object does not contain sample metadata.")
}

# Convert OTU table to matrix for basic checks
otu_matrix <- as(otu_table(physeq), "matrix")

if (!is.numeric(otu_matrix)) {
  stop("OTU/ASV table must be numeric.")
}

if (any(is.na(otu_matrix))) {
  stop("OTU/ASV table contains missing values. Please inspect and clean the count table.")
}

if (any(otu_matrix < 0)) {
  stop("OTU/ASV table contains negative values, which are invalid for Bray-Curtis distance.")
}


############################################################
# 3. Basic QC: sample counts, taxa counts, metadata checks
############################################################

cat("\n==================== BASIC PHYLOSEQ QC ====================\n")
cat("Initial number of samples:", nsamples(physeq), "\n")
cat("Initial number of taxa:", ntaxa(physeq), "\n\n")

cat("Sequencing depth summary before filtering:\n")
print(summary(sample_sums(physeq)))

cat("\nTaxa abundance summary before filtering:\n")
print(summary(taxa_sums(physeq)))

# ---- Remove zero-depth samples and zero-abundance taxa ----
zero_depth_samples <- sample_names(physeq)[sample_sums(physeq) == 0]
if (length(zero_depth_samples) > 0) {
  warning(
    sprintf(
      "Removing %d samples with zero total counts.",
      length(zero_depth_samples)
    )
  )
  physeq <- prune_samples(sample_sums(physeq) > 0, physeq)
}

zero_abundance_taxa <- taxa_names(physeq)[taxa_sums(physeq) == 0]
if (length(zero_abundance_taxa) > 0) {
  warning(
    sprintf(
      "Removing %d taxa with zero total abundance.",
      length(zero_abundance_taxa)
    )
  )
  physeq <- prune_taxa(taxa_sums(physeq) > 0, physeq)
}

cat("\nAfter removing zero-depth samples and zero-abundance taxa:\n")
cat("Number of samples:", nsamples(physeq), "\n")
cat("Number of taxa:", ntaxa(physeq), "\n")

if (nsamples(physeq) < 3) {
  stop("At least 3 samples are required for PCoA.")
}

if (ntaxa(physeq) < 2) {
  stop("At least 2 taxa are required for Bray-Curtis distance.")
}

# ---- Check required sample metadata columns ----
metadata_df <- data.frame(sample_data(physeq), stringsAsFactors = FALSE)

required_metadata_columns <- c(
  "sample_id",
  "subject_id",
  "timepoint",
  "treatment",
  "age",
  "sex"
)

missing_metadata_columns <- setdiff(required_metadata_columns, colnames(metadata_df))

if (length(missing_metadata_columns) > 0) {
  stop(
    sprintf(
      "Missing required sample_data columns: %s",
      paste(missing_metadata_columns, collapse = ", ")
    )
  )
}

# Check uniqueness of sample_id
if (anyDuplicated(metadata_df$sample_id) > 0) {
  warning("sample_id contains duplicated values. Confirm that rownames/sample_names uniquely identify samples.")
}

# Check whether sample_id matches phyloseq sample names
if (!all(as.character(metadata_df$sample_id) %in% sample_names(physeq))) {
  warning(
    "Not all sample_id values are present in phyloseq sample_names. ",
    "The script will use phyloseq sample_names/rownames for alignment."
  )
}

# ---- Remove samples with missing data in variables used for PERMANOVA/plotting ----
model_variables <- c("subject_id", "timepoint", "treatment")

rows_with_missing_model_data <- !complete.cases(metadata_df[, model_variables])

if (any(rows_with_missing_model_data)) {
  warning(
    sprintf(
      "Removing %d samples with missing subject_id, timepoint, or treatment.",
      sum(rows_with_missing_model_data)
    )
  )
  samples_to_keep <- rownames(metadata_df)[!rows_with_missing_model_data]
  physeq <- prune_samples(samples_to_keep, physeq)
  metadata_df <- data.frame(sample_data(physeq), stringsAsFactors = FALSE)
}

# Warn about missing covariates, although they are not included in the default PERMANOVA model
if (any(!complete.cases(metadata_df[, c("age", "sex")]))) {
  warning(
    "Some samples have missing age and/or sex values. ",
    "These variables are not included in the default PERMANOVA model, ",
    "but inspect them if adding covariates."
  )
}

# ---- Convert key variables to factors ----
metadata_df$subject_id <- factor(metadata_df$subject_id)

observed_timepoints <- unique(as.character(metadata_df$timepoint))
unexpected_timepoints <- setdiff(observed_timepoints, expected_timepoint_levels)

if (length(unexpected_timepoints) > 0) {
  warning(
    sprintf(
      "Unexpected timepoint values found: %s. Using observed values as unordered factor levels.",
      paste(unexpected_timepoints, collapse = ", ")
    )
  )
  metadata_df$timepoint <- factor(as.character(metadata_df$timepoint))
} else {
  metadata_df$timepoint <- factor(
    as.character(metadata_df$timepoint),
    levels = expected_timepoint_levels
  )
}

observed_treatments <- unique(as.character(metadata_df$treatment))
unexpected_treatments <- setdiff(observed_treatments, expected_treatment_levels)

if (length(unexpected_treatments) > 0) {
  warning(
    sprintf(
      "Unexpected treatment values found: %s. Using observed values as unordered factor levels.",
      paste(unexpected_treatments, collapse = ", ")
    )
  )
  metadata_df$treatment <- factor(as.character(metadata_df$treatment))
} else {
  metadata_df$treatment <- factor(
    as.character(metadata_df$treatment),
    levels = expected_treatment_levels
  )
}

metadata_df$timepoint <- droplevels(metadata_df$timepoint)
metadata_df$treatment <- droplevels(metadata_df$treatment)

if (nlevels(metadata_df$treatment) < 2) {
  stop("PERMANOVA requires at least two treatment levels.")
}

if (nlevels(metadata_df$timepoint) < 2) {
  stop("PERMANOVA requires at least two timepoint levels.")
}

# Update phyloseq sample_data with cleaned factors
sample_data(physeq) <- sample_data(metadata_df)

cat("\nTreatment x timepoint sample counts:\n")
print(table(metadata_df$treatment, metadata_df$timepoint))

cat("\nNumber of unique subjects:", length(unique(metadata_df$subject_id)), "\n")
cat("Samples per subject summary:\n")
print(summary(table(metadata_df$subject_id)))

# ---- Repeated-measures structure checks ----
samples_per_subject <- table(metadata_df$subject_id)

if (any(samples_per_subject < 2)) {
  warning(
    "Some subjects have fewer than 2 samples. ",
    "Check whether repeated-measures blocking is appropriate for all subjects."
  )
}

treatments_per_subject <- tapply(
  as.character(metadata_df$treatment),
  metadata_df$subject_id,
  function(x) length(unique(x))
)

if (any(treatments_per_subject > 1)) {
  warning(
    "At least one subject_id has more than one treatment assignment. ",
    "Verify whether subject_id is defined correctly."
  )
}

if (all(treatments_per_subject == 1)) {
  message(
    "Treatment appears to be a between-subject variable. ",
    "PERMANOVA will use permutations restricted within subject_id strata, ",
    "as requested. Interpret the treatment main effect cautiously; ",
    "subject-level or split-plot permutation approaches may be preferable for formal treatment inference."
  )
}


############################################################
# 4. Calculate Bray-Curtis distance matrix
############################################################

cat("\n==================== BRAY-CURTIS DISTANCE ====================\n")

# Assumption:
#   Bray-Curtis is calculated on the abundance table currently present in the
#   phyloseq object. If the object contains raw counts, consider whether
#   normalization, rarefaction, or transformation is appropriate for your study.
bray_curtis_distance <- phyloseq::distance(physeq, method = "bray")

if (any(!is.finite(as.matrix(bray_curtis_distance)))) {
  stop("Bray-Curtis distance matrix contains non-finite values.")
}

cat("Bray-Curtis distance matrix dimensions:\n")
print(dim(as.matrix(bray_curtis_distance)))

# Ensure distance labels and metadata rows are aligned
metadata_df <- data.frame(sample_data(physeq), stringsAsFactors = FALSE)
metadata_df <- metadata_df[attr(bray_curtis_distance, "Labels"), , drop = FALSE]

if (!identical(rownames(metadata_df), attr(bray_curtis_distance, "Labels"))) {
  stop("Metadata rows are not aligned with Bray-Curtis distance labels.")
}


############################################################
# 5. Perform PCoA ordination
############################################################

cat("\n==================== PCoA ORDINATION ====================\n")

# stats::cmdscale performs classical multidimensional scaling/PCoA.
# add = TRUE applies an additive correction for non-Euclidean distances,
# which can occur with Bray-Curtis.
pcoa_result <- cmdscale(
  d = bray_curtis_distance,
  k = 2,
  eig = TRUE,
  add = TRUE
)

if (is.null(pcoa_result$points) || ncol(pcoa_result$points) < 2) {
  stop("PCoA failed to return two ordination axes.")
}

positive_eigenvalues <- pcoa_result$eig[pcoa_result$eig > 0]

axis1_percent <- round(100 * pcoa_result$eig[1] / sum(positive_eigenvalues), 1)
axis2_percent <- round(100 * pcoa_result$eig[2] / sum(positive_eigenvalues), 1)

if (!is.null(pcoa_result$ac) && pcoa_result$ac > 0) {
  cat("Additive correction constant used by cmdscale:", pcoa_result$ac, "\n")
}

cat("PCoA axis 1 percent variation:", axis1_percent, "%\n")
cat("PCoA axis 2 percent variation:", axis2_percent, "%\n")

pcoa_coordinates <- as.data.frame(pcoa_result$points)
colnames(pcoa_coordinates) <- c("PCoA1", "PCoA2")
pcoa_coordinates$sample_name <- rownames(pcoa_coordinates)

plot_data <- cbind(
  pcoa_coordinates,
  metadata_df[rownames(pcoa_coordinates), , drop = FALSE]
)


############################################################
# 6. Create publication-quality PCoA plot
############################################################

cat("\n==================== CREATE ORDINATION PLOT ====================\n")

# Color palette: Okabe-Ito-inspired colors for accessibility
base_treatment_colors <- c(
  probiotic = "#0072B2",
  placebo = "#D55E00"
)

treatment_levels <- levels(plot_data$treatment)

if (all(treatment_levels %in% names(base_treatment_colors))) {
  treatment_colors <- base_treatment_colors[treatment_levels]
} else {
  okabe_ito_colors <- c(
    "#E69F00", "#56B4E9", "#009E73", "#F0E442",
    "#0072B2", "#D55E00", "#CC79A7", "#000000"
  )
  if (length(treatment_levels) > length(okabe_ito_colors)) {
    warning("More treatment levels than available manual colors. Colors will be recycled.")
  }
  treatment_colors <- setNames(
    rep(okabe_ito_colors, length.out = length(treatment_levels)),
    treatment_levels
  )
}

timepoint_levels <- levels(plot_data$timepoint)
available_shapes <- c(16, 17, 15, 3, 7, 8)

if (length(timepoint_levels) > length(available_shapes)) {
  warning("More timepoint levels than available manual shapes. Shapes will be recycled.")
}

timepoint_shapes <- setNames(
  rep(available_shapes, length.out = length(timepoint_levels)),
  timepoint_levels
)

ordination_plot <- ggplot(
  plot_data,
  aes(
    x = PCoA1,
    y = PCoA2,
    color = treatment,
    shape = timepoint
  )
) +
  geom_point(size = 3.0, alpha = 0.85, stroke = 0.8) +
  scale_color_manual(values = treatment_colors) +
  scale_shape_manual(values = timepoint_shapes) +
  labs(
    title = "PCoA of microbiome composition",
    subtitle = "Bray-Curtis dissimilarity",
    x = paste0("PCoA1 (", axis1_percent, "%)"),
    y = paste0("PCoA2 (", axis2_percent, "%)"),
    color = "Treatment",
    shape = "Timepoint"
  ) +
  theme_bw(base_size = 13) +
  theme(
    plot.title = element_text(face = "bold", hjust = 0),
    plot.subtitle = element_text(hjust = 0),
    panel.grid.major = element_line(color = "grey90", linewidth = 0.25),
    panel.grid.minor = element_blank(),
    axis.title = element_text(face = "bold"),
    legend.title = element_text(face = "bold"),
    legend.position = "right"
  ) +
  guides(
    color = guide_legend(override.aes = list(size = 4, alpha = 1)),
    shape = guide_legend(override.aes = list(size = 4, alpha = 1))
  )

# Add treatment ellipses only when each treatment has enough samples
if (all(table(plot_data$treatment) >= 3)) {
  ordination_plot <- ordination_plot +
    stat_ellipse(
      aes(group = treatment, color = treatment),
      type = "t",
      level = 0.68,
      linewidth = 0.7,
      linetype = "dashed",
      show.legend = FALSE
    )
}

ggsave(
  filename = output_plot_path,
  plot = ordination_plot,
  device = "pdf",
  width = 7,
  height = 5.5,
  units = "in"
)

cat("PCoA plot saved to:", output_plot_path, "\n")


############################################################
# 7. PERMANOVA: treatment x timepoint with subject blocking
############################################################

cat("\n==================== PERMANOVA ====================\n")

# Model:
#   Bray-Curtis distance ~ treatment * timepoint
#
# Repeated measures:
#   strata = subject_id restricts permutations within subjects.
#
# Important note:
#   If treatment is constant within each subject, treatment is a between-subject
#   factor. Blocking permutations within subjects is appropriate for repeated
#   measurements but can make treatment main-effect inference conservative or
#   not fully exchangeable. The treatment:timepoint interaction is often the
#   primary longitudinal intervention contrast.
permanova_result <- vegan::adonis2(
  bray_curtis_distance ~ treatment * timepoint,
  data = metadata_df,
  permutations = permanova_permutations,
  by = "terms",
  strata = metadata_df$subject_id
)

cat("\nPERMANOVA formula:\n")
cat("Bray-Curtis distance ~ treatment * timepoint\n")

cat("\nPermutation scheme:\n")
cat(
  permanova_permutations,
  "permutations restricted within subject_id strata\n"
)

cat("\nPERMANOVA results:\n")
print(permanova_result)

cat("\nNote: The treatment:timepoint row tests the longitudinal interaction.\n")
cat("Consider checking homogeneity of dispersion with vegan::betadisper.\n")

############################################################
# End of script
