# ==============================================================================
# R SCRIPT FOR PROBIOTIC INTERVENTION STUDY: ORDINATION & PERMANOVA
#
# Author: AI Assistant for Microbiome Researcher
# Date: 2023-10-27
#
# Description:
# This script loads a phyloseq object, performs PCoA on Bray-Curtis distances,
# visualizes the results, and runs a PERMANOVA to test the effects of
# treatment and timepoint on microbial community structure, accounting for
# repeated measures from subjects.
# ==============================================================================


# ------------------------------------------------------------------------------
# SECTION 0: SETUP - LOAD LIBRARIES AND DEFINE PATHS
# ------------------------------------------------------------------------------

# Install packages if they are not already installed
# Note: phyloseq is a Bioconductor package
# if (!requireNamespace("BiocManager", quietly = TRUE))
#     install.packages("BiocManager")
# BiocManager::install("phyloseq")
# install.packages(c("vegan", "ggplot2", "dplyr"))

# Load required libraries
library(phyloseq)
library(vegan)
library(ggplot2)
library(dplyr) # Used for data manipulation, part of the tidyverse

# --- Define input and output paths ---
# Assumption: Your R session is running in the project's root directory.
input_phyloseq_path <- "data/probiotic_study.rds"
output_plot_path <- "results/pcoa_bray_curtis_plot.pdf"

# Create results directory if it doesn't exist
if (!dir.exists("results")) {
  dir.create("results")
}


# ------------------------------------------------------------------------------
# SECTION 1: LOAD AND INSPECT PHYLOSEQ OBJECT
# ------------------------------------------------------------------------------

# --- Load data ---
# Basic error handling: check if the input file exists
if (!file.exists(input_phyloseq_path)) {
  stop("Input file not found at: ", input_phyloseq_path, 
       "\nPlease check the file path.")
}
ps_object <- readRDS(input_phyloseq_path)

# --- Basic QC and Inspection ---
cat("--- Phyloseq Object Summary ---\n")
print(ps_object)
cat("\n")

# Check sample and taxa counts
cat("Number of samples:", nsamples(ps_object), "\n")
cat("Number of taxa (OTUs/ASVs):", ntaxa(ps_object), "\n\n")

# Inspect the sample data to ensure columns are correct
cat("--- Sample Data Head ---\n")
sample_df <- data.frame(sample_data(ps_object))
print(head(sample_df))
cat("\n")

# --- Data Validation ---
# Check for required columns
required_cols <- c("subject_id", "timepoint", "treatment")
if (!all(required_cols %in% colnames(sample_df))) {
  stop("Sample data is missing one or more required columns: ", 
       paste(required_cols, collapse = ", "))
}

# Check for missing values in key variables
if (any(is.na(sample_df$treatment)) || any(is.na(sample_df$timepoint)) || any(is.na(sample_df$subject_id))) {
  warning("Missing values detected in key columns (treatment, timepoint, or subject_id). This may affect analysis.")
}

# Ensure key variables are factors for plotting and analysis
# This helps control the order and appearance in plots/stats
sample_data(ps_object)$timepoint <- factor(sample_data(ps_object)$timepoint, 
                                           levels = c("baseline", "week4", "week8"))
sample_data(ps_object)$treatment <- factor(sample_data(ps_object)$treatment,
                                           levels = c("placebo", "probiotic"))


# ------------------------------------------------------------------------------
# SECTION 2: CALCULATE BRAY-CURTIS DISTANCE
# ------------------------------------------------------------------------------
# Assumption: The OTU table contains raw counts. For beta diversity, it's
# common to work with relative abundance to account for sequencing depth variation.
# We will transform counts to relative abundance before calculating distance.

# --- Transform counts to relative abundance ---
ps_rel_abund <- transform_sample_counts(ps_object, function(x) x / sum(x))

# --- Calculate Bray-Curtis distance matrix ---
# This is a measure of community dissimilarity.
bray_dist <- phyloseq::distance(ps_rel_abund, method = "bray")

# Alternative: One could use vegan's vegdist() directly on the OTU table:
# otu_table <- as(otu_table(ps_rel_abund), "matrix")
# bray_dist_vegan <- vegan::vegdist(otu_table, method = "bray")


# ------------------------------------------------------------------------------
# SECTION 3: PERFORM PRINCIPAL COORDINATES ANALYSIS (PCoA)
# ------------------------------------------------------------------------------
# PCoA (also known as MDS) is a method to visualize the dissimilarities
# between samples in a low-dimensional space.

pcoa_results <- ordinate(ps_rel_abund, method = "PCoA", distance = bray_dist)

# Check the variance explained by the first few axes
pcoa_variance_explained <- pcoa_results$values$Relative_eig
cat("--- PCoA Variance Explained ---\n")
cat("Axis 1:", round(pcoa_variance_explained[1] * 100, 2), "%\n")
cat("Axis 2:", round(pcoa_variance_explained[2] * 100, 2), "%\n\n")


# ------------------------------------------------------------------------------
# SECTION 4: CREATE AND SAVE ORDINATION PLOT
# ------------------------------------------------------------------------------
# We will use ggplot2 for a publication-quality figure.
# The plot will show samples colored by treatment and shaped by timepoint.

pcoa_plot <- plot_ordination(
  physeq = ps_rel_abund,
  ordination = pcoa_results,
  color = "treatment",
  shape = "timepoint"
) +
  geom_point(size = 3, alpha = 0.75) + # Set point size and transparency
  scale_color_manual(values = c("placebo" = "#1f78b4", "probiotic" = "#e31a1c")) + # Custom colors
  scale_shape_manual(values = c("baseline" = 16, "week4" = 17, "week8" = 15)) + # Custom shapes
  labs(
    title = "PCoA of Gut Microbiome Composition",
    subtitle = "Bray-Curtis Dissimilarity",
    x = paste0("PCo1 (", round(pcoa_variance_explained[1] * 100, 2), "%)"),
    y = paste0("PCo2 (", round(pcoa_variance_explained[2] * 100, 2), "%)"),
    color = "Treatment Group",
    shape = "Timepoint"
  ) +
  theme_bw() + # A clean theme
  theme(
    plot.title = element_text(hjust = 0.5, face = "bold"),
    plot.subtitle = element_text(hjust = 0.5),
    legend.position = "right"
  ) +
  stat_ellipse(aes(group = treatment), type = "t", linetype
