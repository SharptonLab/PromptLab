# Microbiome Ordination Analysis Script

```r
# =============================================================================
# Probiotic Study: PCoA Ordination and PERMANOVA Analysis
# =============================================================================
# Purpose: Visualize community composition and test treatment x timepoint
#          effects accounting for repeated measures (subject as block).
# Input:   data/probiotic_study.rds (phyloseq object)
# Output:  results/pcoa_plot.pdf, PERMANOVA summary to console
# =============================================================================

# ---- 0. Setup ---------------------------------------------------------------
suppressPackageStartupMessages({
  library(phyloseq)   # microbiome data container, ordination wrapper
  library(vegan)      # Bray-Curtis distance, PERMANOVA (adonis2)
  library(ggplot2)    # plotting
})

# Reproducibility (PERMANOVA uses permutations)
set.seed(42)

# Paths
input_path  <- "data/probiotic_study.rds"
output_dir  <- "results"
plot_path   <- file.path(output_dir, "pcoa_plot.pdf")

if (!dir.exists(output_dir)) dir.create(output_dir, recursive = TRUE)

# ---- 1. Load phyloseq object and basic QC -----------------------------------
if (!file.exists(input_path)) {
  stop("Input file not found: ", input_path)
}

ps <- readRDS(input_path)

if (!inherits(ps, "phyloseq")) {
  stop("Loaded object is not a phyloseq object. Class: ", class(ps))
}

# Required sample metadata columns
required_cols <- c("sample_id", "subject_id", "timepoint", "treatment",
                   "age", "sex")
sample_df <- as(sample_data(ps), "data.frame")
missing_cols <- setdiff(required_cols, colnames(sample_df))
if (length(missing_cols) > 0) {
  stop("Missing required sample_data columns: ",
       paste(missing_cols, collapse = ", "))
}

# Coerce key variables to factors with expected levels
# Assumption: timepoint levels are exactly "baseline", "week4", "week8"
sample_df$timepoint <- factor(sample_df$timepoint,
                              levels = c("baseline", "week4", "week8"))
sample_df$treatment <- factor(sample_df$treatment,
                              levels = c("placebo", "probiotic"))
sample_df$subject_id <- as.factor(sample_df$subject_id)

# Check for NAs in key variables
key_vars <- c("timepoint", "treatment", "subject_id")
na_counts <- sapply(sample_df[key_vars], function(x) sum(is.na(x)))
if (any(na_counts > 0)) {
  warning("NAs detected in key variables: ",
          paste(names(na_counts)[na_counts > 0], collapse = ", "))
}

# Write back cleaned sample_data
sample_data(ps) <- sample_data(sample_df)

# QC summary
cat("------ QC Summary ------\n")
cat("Number of samples:", nsamples(ps), "\n")
cat("Number of taxa:   ", ntaxa(ps), "\n")
cat("Sample sums (sequencing depth) summary:\n")
print(summary(sample_sums(ps)))
cat("\nSamples per treatment x timepoint:\n")
print(table(sample_df$treatment, sample_df$timepoint))

# Drop samples with zero counts (would break Bray-Curtis)
zero_samples <- sample_names(ps)[sample_sums(ps) == 0]
if (length(zero_samples) > 0) {
  message("Dropping ", length(zero_samples), " samples with zero counts.")
  ps <- prune_samples(sample_sums(ps) > 0, ps)
}

# ---- 2. Normalize counts ----------------------------------------------------
# Convert to relative abundance for Bray-Curtis.
# Alternative: rarefy_even_depth() if depth varies substantially.
ps_rel <- transform_sample_counts(ps, function(x) x / sum(x))

# ---- 3. Bray-Curtis distance ------------------------------------------------
bc_dist <- phyloseq::distance(ps_rel, method = "bray")

# ---- 4. PCoA ordination -----------------------------------------------------
pcoa_ord <- ordinate(ps_rel, method = "PCoA", distance = bc_dist)

# Variance explained by first two axes
eig_vals <- pcoa_ord$values$Relative_eig
pc1_var <- round(100 * eig_vals[1], 1)
pc2_var <- round(100 * eig_vals[2], 1)

# ---- 5. Ordination plot -----------------------------------------------------
plot_df <- plot_ordination(ps_rel, pcoa_ord,
                           color = "treatment",
                           shape = "timepoint",
                           justDF = TRUE)

pcoa_plot <- ggplot(plot_df, aes(x = Axis.1, y = Axis.2,
                                 color = treatment, shape = timepoint)) +
  geom_point(size = 3, alpha = 0.8) +
  stat_ellipse(aes(group = interaction(treatment, timepoint)),
               type = "norm", level = 0.68, linewidth = 0.4) +
  scale_color_manual(values = c(placebo = "#377EB8", probiotic = "#E41A1C")) +
  scale_shape_manual(values = c(baseline = 16, week4 = 17, week8 = 15)) +
  labs(x = paste0("PCoA1 (", pc1_var, "%)"),
       y = paste0("PCoA2 (", pc2_var, "%)"),
       title = "PCoA of Bray-Curtis distances",
       color = "Treatment", shape = "Timepoint") +
  theme_bw(base_size = 12) +
  theme(panel.grid.minor = element_blank(),
        legend.position = "right",
        plot.title = element_text(face = "bold"))

ggsave(plot_path, pcoa_plot, width = 7, height = 5.5, device = "pdf")
cat("\nOrdination plot saved to:", plot_path, "\n")

# ---- 6. PERMANOVA with repeated measures ------------------------------------
# Strategy: restrict permutations within subjects using how() with blocks =
# subject_id. This respects the repeated-measures structure: timepoints are
# permuted within each subject, so the treatment x timepoint interaction is
# tested while controlling for subject-level non-independence.
#
# Alternative approaches:
#   - strata = subject_id (older adonis()) — equivalent for blocking.
#   - Include subject_id as a fixed effect term (not ideal: collinear w/ treatment
#     since treatment is between-subject).
#   - Mixed-effects ordination (e.g., GLLVM) for subject random effects.

meta_df <- as(sample_data(ps_rel), "data.frame")

# Align distance matrix and metadata
stopifnot(all(labels(bc_dist) == rownames(meta_df)))

perm_control <- how(nperm = 999,
                    blocks = meta_df$subject_id)

permanova_res <- adonis2(bc_dist ~ treatment * timepoint,
                         data        = meta_df,
                         permutations = perm_control,
                         by          = "terms")

cat("\n------ PERMANOVA (Bray-Curtis, permutations blocked by subject_id) ------\n")
print(permanova_res)

# ---- 7. Dispersion check (assumption of PERMANOVA) --------------------------
# Significant PERMANOVA can reflect dispersion differences rather than
# centroid shifts; report betadisper for transparency.
group_var <- interaction(meta_df$treatment, meta_df$timepoint)
disp_res  <- betadisper(bc_dist, group_var)
disp_test <- permutest(disp_res, permutations = 999)

cat("\n------ Beta-dispersion test (treatment x timepoint groups) ------\n")
print(disp_test)

# ---- Done -------------------------------------------------------------------
cat("\nAnalysis complete.\n")
```
