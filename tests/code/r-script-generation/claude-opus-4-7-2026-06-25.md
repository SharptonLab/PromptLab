# Microbiome Ordination Analysis Script

```r
# =============================================================================
# Probiotic Study: PCoA Ordination and PERMANOVA Analysis
# =============================================================================
# Purpose: Perform Bray-Curtis PCoA and PERMANOVA on probiotic intervention data
# Input:   phyloseq object (data/probiotic_study.rds)
# Output:  PCoA ordination PDF + PERMANOVA results to console
# =============================================================================

# ---- 1. Setup: Load packages -----------------------------------------------
# Core packages requested: phyloseq, vegan, ggplot2
# Additional: dplyr (data manipulation), tibble (cleaner output)
# Both are tidyverse/CRAN and improve maintainability per requirements.

suppressPackageStartupMessages({
  library(phyloseq)
  library(vegan)
  library(ggplot2)
  library(dplyr)
  library(tibble)
})

# Set seed for reproducibility (PERMANOVA uses permutations)
set.seed(20240101)

# Define file paths
input_path  <- "data/probiotic_study.rds"
output_pdf  <- "results/pcoa_ordination.pdf"

# Create output directory if it doesn't exist
if (!dir.exists("results")) dir.create("results", recursive = TRUE)


# ---- 2. Load data and basic QC ---------------------------------------------

# Check input file exists
if (!file.exists(input_path)) {
  stop("Input file not found: ", input_path)
}

ps <- readRDS(input_path)

# Confirm we have a phyloseq object
if (!inherits(ps, "phyloseq")) {
  stop("Input is not a phyloseq object. Got class: ", class(ps))
}

# Basic QC summary
cat("=== Phyloseq Object QC ===\n")
cat("Number of samples:     ", nsamples(ps), "\n")
cat("Number of taxa:        ", ntaxa(ps), "\n")
cat("Sample variables:      ", paste(sample_variables(ps), collapse = ", "), "\n")
cat("Taxa ranks:            ", paste(rank_names(ps), collapse = ", "), "\n\n")

# Verify required metadata columns are present
required_vars <- c("sample_id", "subject_id", "timepoint", "treatment")
missing_vars  <- setdiff(required_vars, sample_variables(ps))
if (length(missing_vars) > 0) {
  stop("Missing required sample variables: ", paste(missing_vars, collapse = ", "))
}

# Coerce key variables to factors with explicit levels
# ASSUMPTION: timepoint values are exactly "baseline", "week4", "week8"
# ASSUMPTION: treatment values are exactly "probiotic", "placebo"
sample_data(ps)$timepoint <- factor(sample_data(ps)$timepoint,
                                    levels = c("baseline", "week4", "week8"))
sample_data(ps)$treatment <- factor(sample_data(ps)$treatment,
                                    levels = c("placebo", "probiotic"))
sample_data(ps)$subject_id <- as.factor(sample_data(ps)$subject_id)

# Check for missing values in key variables
meta_df <- as(sample_data(ps), "data.frame")
if (any(is.na(meta_df[, required_vars]))) {
  warning("NA values detected in required metadata columns. Removing affected samples.")
  keep_samples <- complete.cases(meta_df[, required_vars])
  ps <- prune_samples(keep_samples, ps)
}

# Check for samples with zero counts (would break Bray-Curtis)
sample_sums_vec <- sample_sums(ps)
if (any(sample_sums_vec == 0)) {
  warning("Removing ", sum(sample_sums_vec == 0), " samples with zero total counts.")
  ps <- prune_samples(sample_sums_vec > 0, ps)
}

# Cross-tabulate design to verify balance
cat("=== Design balance (treatment x timepoint) ===\n")
print(table(sample_data(ps)$treatment, sample_data(ps)$timepoint))
cat("\n")


# ---- 3. Normalize counts ---------------------------------------------------
# Convert to relative abundance for Bray-Curtis.
# ALTERNATIVE: rarefaction (rarefy_even_depth) or CSS/DESeq2 normalization.
# Relative abundance is a defensible default for community-level beta diversity.

ps_rel <- transform_sample_counts(ps, function(x) x / sum(x))


# ---- 4. Calculate Bray-Curtis distance and perform PCoA --------------------

# Compute Bray-Curtis distance using phyloseq (wraps vegan::vegdist)
bray_dist <- phyloseq::distance(ps_rel, method = "bray")

# Perform PCoA (Principal Coordinates Analysis)
pcoa_ord <- ordinate(ps_rel, method = "PCoA", distance = bray_dist)

# Extract variance explained by first two axes
eig_vals    <- pcoa_ord$values$Relative_eig
pc1_var_pct <- round(100 * eig_vals[1], 1)
pc2_var_pct <- round(100 * eig_vals[2], 1)

cat("=== PCoA Variance Explained ===\n")
cat("Axis 1: ", pc1_var_pct, "%\n")
cat("Axis 2: ", pc2_var_pct, "%\n\n")


# ---- 5. Build ordination plot ----------------------------------------------

# Use phyloseq's plot_ordination, then enhance with ggplot2
pcoa_plot <- plot_ordination(ps_rel, pcoa_ord,
                             color = "treatment",
                             shape = "timepoint") +
  geom_point(size = 3, alpha = 0.8) +
  stat_ellipse(aes(group = interaction(treatment, timepoint)),
               type = "norm", linetype = 2, alpha = 0.6) +
  scale_color_manual(values = c("placebo" = "#377EB8", "probiotic" = "#E41A1C")) +
  scale_shape_manual(values = c("baseline" = 16, "week4" = 17, "week8" = 15)) +
  labs(
    title    = "PCoA of Gut Microbiome (Bray-Curtis)",
    subtitle = "Probiotic vs Placebo across timepoints",
    x        = paste0("PCoA Axis 1 (", pc1_var_pct, "%)"),
    y        = paste0("PCoA Axis 2 (", pc2_var_pct, "%)"),
    color    = "Treatment",
    shape    = "Timepoint"
  ) +
  theme_bw(base_size = 12) +
  theme(
    panel.grid.minor = element_blank(),
    legend.position  = "right",
    plot.title       = element_text(face = "bold")
  )

# Save publication-quality PDF
ggsave(output_pdf, pcoa_plot, width = 8, height = 6, device = cairo_pdf)
cat("Ordination plot saved to: ", output_pdf, "\n\n")


# ---- 6. PERMANOVA with repeated-measures handling --------------------------
# Use adonis2 with strata = subject_id to constrain permutations within subjects,
# accounting for the repeated-measures design (3 timepoints per subject).
#
# NOTE: 'strata' restricts permutations so samples are only shuffled within
# subjects. This is the standard vegan approach for blocked designs.
# ALTERNATIVE: use the 'how()' permutation design from vegan for more control,
# or PERMANOVA via the 'pairwiseAdonis' package for post-hoc tests.

meta_df <- as(sample_data(ps_rel), "data.frame")

# Define permutation scheme: free permutation within each subject
perm_design <- with(meta_df,
                    how(blocks = subject_id, nperm = 999))

# Fit PERMANOVA model with interaction
permanova_res <- adonis2(
  bray_dist ~ treatment * timepoint,
  data        = meta_df,
  permutations = perm_design,
  by          = "terms"   # sequential SS; use "margin" for type III-like SS
)

cat("=== PERMANOVA Results (Bray-Curtis) ===\n")
cat("Model: distance ~ treatment * timepoint\n")
cat("Permutations blocked by subject_id (n = 999)\n\n")
print(permanova_res)

# Save PERMANOVA results as tidy CSV for downstream reporting
permanova_df <- permanova_res %>%
  as.data.frame() %>%
  rownames_to_column("term")
write.csv(permanova_df, "results/permanova_results.csv", row.names = FALSE)
cat("\nPERMANOVA table saved to: results/permanova_results.csv\n")


# ---- 7. Beta-dispersion check (assumption diagnostic) ----------------------
# PERMANOVA assumes similar within-group dispersion. If groups differ in spread,
# significant PERMANOVA could reflect dispersion rather than centroid shifts.

group_var <- interaction(meta_df$treatment, meta_df$timepoint)
beta_disp <- betadisper(bray_dist, group_var)
disp_test <- permutest(beta_disp, permutations = 999)

cat("\n=== Beta-dispersion (homogeneity check) ===\n")
print(disp_test)

cat("\n=== Analysis complete ===\n")
```
