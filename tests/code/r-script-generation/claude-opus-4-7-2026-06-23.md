# =============================================================================
# Microbiome Ordination Analysis: Probiotic Intervention Study
# Author: [Your Name]
# Purpose: PCoA visualization and PERMANOVA testing of community composition
#          across treatment (probiotic vs placebo) and timepoint (BL/W4/W8)

# -----------------------------------------------------------------------------
# 1. Setup: Load packages
# -----------------------------------------------------------------------------
# phyloseq: microbiome data container and ordination wrappers
# vegan:    distance matrices, PERMANOVA (adonis2), dispersion tests
# ggplot2:  publication-quality plotting
# dplyr:    light-weight data manipulation (tidyverse subset)

suppressPackageStartupMessages({
  library(phyloseq)
  library(vegan)
  library(ggplot2)
  library(dplyr)
})

set.seed(42)  # reproducibility for permutation tests

# -----------------------------------------------------------------------------
# 2. Load data and basic QC
# -----------------------------------------------------------------------------
input_path  <- "data/probiotic_study.rds"
output_dir  <- "results"
if (!dir.exists(output_dir)) dir.create(output_dir, recursive = TRUE)

# Confirm file exists before attempting to load
if (!file.exists(input_path)) {
  stop("Input file not found at: ", input_path)
}

ps <- readRDS(input_path)

# Validate object class
if (!inherits(ps, "phyloseq")) {
  stop("Loaded object is not of class 'phyloseq'. Got: ", class(ps))
}

# Report basic QC metrics
cat("========== Phyloseq Object QC ==========\n")
cat("Number of samples : ", nsamples(ps), "\n")
cat("Number of taxa    : ", ntaxa(ps), "\n")
cat("Sample variables  : ", paste(sample_variables(ps), collapse = ", "), "\n")
cat("Library size range: ", paste(range(sample_sums(ps)), collapse = " - "), "\n\n")

# Check required metadata columns are present
required_vars <- c("sample_id", "subject_id", "timepoint", "treatment")
missing_vars  <- setdiff(required_vars, sample_variables(ps))
if (length(missing_vars) > 0) {
  stop("Missing required sample variables: ", paste(missing_vars, collapse = ", "))
}

# Flag samples with very low read counts (common QC threshold: 1000 reads)
low_depth_samples <- sum(sample_sums(ps) < 1000)
if (low_depth_samples > 0) {
  warning(low_depth_samples, " samples have <1000 reads. Consider filtering.")
}

# Ensure factors with intended ordering (timepoint should be ordered logically)
sample_data(ps)$timepoint <- factor(sample_data(ps)$timepoint,
                                    levels = c("baseline", "week4", "week8"))
sample_data(ps)$treatment <- factor(sample_data(ps)$treatment,
                                    levels = c("placebo", "probiotic"))

# -----------------------------------------------------------------------------
# 3. Normalization
# -----------------------------------------------------------------------------
# ASSUMPTION: counts are raw. Convert to relative abundance for Bray-Curtis.
# Alternative: rarefy to even depth (rarefy_even_depth) or use CSS/DESeq2
#              normalization. Relative abundance is a common, simple choice
#              for compositional ordination with Bray-Curtis.

ps_rel <- transform_sample_counts(ps, function(x) x / sum(x))

# Remove low-prevalence taxa to reduce noise (present in <5% of samples)
min_prev   <- 0.05 * nsamples(ps_rel)
prevalence <- apply(otu_table(ps_rel), 
                    ifelse(taxa_are_rows(ps_rel), 1, 2),
                    function(x) sum(x > 0))
ps_rel     <- prune_taxa(prevalence >= min_prev, ps_rel)

cat("Taxa remaining after prevalence filter (>=5% samples): ", ntaxa(ps_rel), "\n\n")

# -----------------------------------------------------------------------------
# 4. Bray-Curtis distance and PCoA ordination
# -----------------------------------------------------------------------------
bc_dist <- phyloseq::distance(ps_rel, method = "bray")

pcoa_ord <- ordinate(ps_rel, method = "PCoA", distance = bc_dist)

# Extract eigenvalue-derived variance explained for axis labels
eig_pct <- round(100 * pcoa_ord$values$Relative_eig[1:2], 1)

# -----------------------------------------------------------------------------
# 5. Visualization: PCoA colored by treatment, shaped by timepoint
# -----------------------------------------------------------------------------
pcoa_plot <- plot_ordination(ps_rel, pcoa_ord,
                             color = "treatment",
                             shape = "timepoint") +
  geom_point(size = 3, alpha = 0.85) +
  stat_ellipse(aes(group = treatment), type = "norm", linetype = 2, size = 0.4) +
  scale_color_manual(values = c("placebo" = "#1f77b4", "probiotic" = "#d62728")) +
  scale_shape_manual(values = c("baseline" = 16, "week4" = 17, "week8" = 15)) +
  labs(
    title    = "PCoA of Bray-Curtis Distances",
    subtitle = "Probiotic Intervention Study",
    x        = paste0("PCoA 1 (", eig_pct[1], "%)"),
    y        = paste0("PCoA 2 (", eig_pct[2], "%)"),
    color    = "Treatment",
    shape    = "Timepoint"
  ) +
  theme_bw(base_size = 12) +
  theme(
    panel.grid.minor = element_blank(),
    legend.position  = "right",
    plot.title       = element_text(face = "bold")
  )

# Save publication-quality PDF
pdf_path <- file.path(output_dir, "pcoa_braycurtis.pdf")
ggsave(pdf_path, pcoa_plot, width = 7, height = 5.5, device = cairo_pdf)
cat("Saved ordination plot to: ", pdf_path, "\n\n")

# -----------------------------------------------------------------------------
# 6. PERMANOVA: treatment × timepoint with subject as blocking factor
# -----------------------------------------------------------------------------
# Repeated measures handling:
#   - Use 'strata' / 'permutations = how(blocks = subject_id)' so permutations
#     occur WITHIN subjects, respecting non-independence across timepoints.
# Alternative: PERMANOVA on baseline-adjusted distances, or mixed-effects
#              dissimilarity models (e.g., MDMR). Blocked permutation is the
#              most common approach for this design.

meta_df <- data.frame(sample_data(ps_rel))

# Build permutation design: free permutation within subject, none across
perm_design <- with(meta_df, how(nperm = 999,
                                 blocks = factor(subject_id)))

permanova_res <- adonis2(
  bc_dist ~ treatment * timepoint,
  data         = meta_df,
  permutations = perm_design,
  by           = "terms"
)

cat("========== PERMANOVA Results ==========\n")
print(permanova_res)
cat("\n")

# -----------------------------------------------------------------------------
# 7. Dispersion check (PERMDISP) — guards against confounding by group spread
# -----------------------------------------------------------------------------
disp_treatment <- betadisper(bc_dist, meta_df$treatment)
disp_timepoint <- betadisper(bc_dist, meta_df$timepoint)

cat("========== Beta-dispersion (PERMDISP) ==========\n")
cat("--- Treatment ---\n");  print(permutest(disp_treatment, permutations = 999))
cat("--- Timepoint ---\n");  print(permutest(disp_timepoint, permutations = 999))

# -----------------------------------------------------------------------------
# 8. Save results
# -----------------------------------------------------------------------------
results_path <- file.path(output_dir, "permanova_results.txt")
sink(results_path)
cat("PERMANOVA: Bray-Curtis ~ treatment * timepoint\n")
cat("Permutations blocked within subject_id (n = 999)\n\n")
print(permanova_res)
cat("\n\n--- Beta-dispersion (treatment) ---\n")
print(permutest(disp_treatment, permutations = 999))
cat("\n--- Beta-dispersion (timepoint) ---\n")
print(permutest(disp_timepoint, permutations = 999))
sink()

cat("Saved PERMANOVA summary to: ", results_path, "\n")
cat("Analysis complete.\n")
