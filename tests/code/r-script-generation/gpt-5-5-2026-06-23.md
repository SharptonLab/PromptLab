# Required packages

if (length(missing_packages) > 0) {

library(phyloseq)

input_rds_path <- "data/probiotic_study.rds"

# Create output directories if they do not already exist
dir.create(dirname(figure_output_path), recursive = TRUE, showWarnings = FALSE)
dir.create(dirname(permanova_output_path), recursive = TRUE, showWarnings = FALSE)

set.seed(12345)

# Check that input file exists
if (!file.exists(input_rds_path)) {

# Load RDS file
physeq <- readRDS(input_rds_path)

# Confirm object is a phyloseq object
if (!inherits(physeq, "phyloseq")) {
  stop("The loaded object is not a phyloseq object.")
}

cat("\nSuccessfully loaded phyloseq object from:", input_rds_path, "\n")


############################################################
# 3. Basic QC checks
############################################################

cat("\n==================== Basic QC ====================\n")

# Check that essential components are present
if (is.null(otu_table(physeq, errorIfNULL = FALSE))) {
  stop("The phyloseq object does not contain an OTU/ASV table.")
}

if (is.null(sample_data(physeq, errorIfNULL = FALSE))) {
  stop("The phyloseq object does not contain sample metadata.")
}

if (is.null(tax_table(physeq, errorIfNULL = FALSE))) {
  warning("The phyloseq object does not contain a taxonomy table.")
}

# Print sample and taxa counts
cat("Number of samples:", nsamples(physeq), "\n")
cat("Number of taxa:", ntaxa(physeq), "\n")

# Extract OTU table as matrix
otu_matrix <- as(otu_table(physeq), "matrix")

# Ensure taxa are rows for matrix-level QC
if (!taxa_are_rows(physeq)) {

}

# Check for missing or negative abundance values
if (any(is.na(otu_matrix))) {
  stop("OTU/ASV table contains missing values. Please resolve before analysis.")
}

if (any(otu_matrix < 0)) {
  stop("OTU/ASV table contains negative values, which are not valid for Bray-Curtis distance.")
}

# Calculate library sizes and taxa totals
sample_library_sizes <- sample_sums(physeq)
taxa_total_abundances <- taxa_sums(physeq)

cat("\nZero-count samples:", length(zero_count_samples), "\n")
cat("Zero-count taxa:", length(zero_count_taxa), "\n")

if (length(zero_count_samples) > 0) {

if (length(zero_count_taxa) > 0) {
  warning("Removing ", length(zero_count_taxa), " zero-count taxa.")

cat("\nAfter pruning:\n")
cat("Number of samples:", nsamples(physeq), "\n")
cat("Number of taxa:", ntaxa(physeq), "\n")

# 4. Check and prepare sample metadata
############################################################

# Convert sample metadata to data.frame
sample_metadata <- as(sample_data(physeq), "data.frame")

# Required metadata columns
required_metadata_columns <- c(
  "sample_id",

  "timepoint",
  "treatment",
  "age",

cat("\n==================== PERMANOVA ====================\n")

# Prepare metadata in the same order as the Bray-Curtis distance matrix
distance_sample_names <- labels(bray_distance)

permanova_metadata <- sample_metadata[distance_sample_names, , drop = FALSE]

# Check alignment between distance matrix and metadata
if (!identical(rownames(permanova_metadata), distance_sample_names)) {
  stop("Sample order mismatch between distance matrix and metadata.")
}

# PERMANOVA model:
#   Bray-Curtis distance ~ treatment * timepoint
#
# Repeated measures:
#   strata = subject_id constrains permutations within subjects.
#
# Important assumption:
#   If treatment is assigned between subjects and does not vary within subject,
#   subject-level blocking is appropriate for testing within-subject timepoint
#   effects and treatment-by-timepoint interaction, but the treatment main effect
#   may not be fully testable under within-subject-only permutations.
permanova_result <- vegan::adonis2(
  bray_distance ~ treatment * timepoint,
  data = permanova_metadata,
  permutations = 9999,
  strata = permanova_metadata$subject_id,
  by = "terms"
)

cat("\nPERMANOVA results: Bray-Curtis ~ treatment * timepoint\n")
cat("Permutations constrained within subject_id strata.\n\n")
print(permanova_result)

# Save PERMANOVA summary to text file
sink(permanova_output_path)
cat("PERMANOVA results: Bray-Curtis ~ treatment * timepoint\n")
cat("Distance: Bray-Curtis\n")
cat("Permutations: 9999\n")
cat("Permutation strata: subject_id\n")
cat("Terms tested sequentially using by = 'terms'\n\n")
print(permanova_result)
sink()

cat("\nPERMANOVA summary saved to:", permanova_output_path, "\n")


############################################################
# 9. Optional diagnostic: homogeneity of dispersion
############################################################

cat("\n==================== Dispersion Diagnostic ====================\n")

# PERMANOVA can be sensitive to differences in within-group dispersion.
# This diagnostic tests whether dispersion differs by treatment-timepoint group.
permanova_metadata$treatment_timepoint <- interaction(
  permanova_metadata$treatment,
  permanova_metadata$timepoint,
  drop = TRUE
)

dispersion_result <- vegan::betadisper(
  bray_distance,
  group = permanova_metadata$treatment_timepoint
)

dispersion_anova <- anova(dispersion_result)

cat("\nTest for homogeneity of multivariate dispersion by treatment-timepoint group:\n")
print(dispersion_anova)

cat("\nAnalysis complete.\n")
