################################################################################
#
# R Script for Microbiome Community Analysis (PCoA, PERMANOVA)
#
# Author: [Your Name/Lab]
# Date: [Current Date]
#
# Description: This script loads a phyloseq object, performs beta diversity
#              analysis using Bray-Curtis distance and PCoA, visualizes the
#              ordination, and runs a PERMANOVA to test for effects of
#              treatment and timepoint while accounting for repeated measures.
#
################################################################################


# --- Section 0: Setup and Configuration ---

# Load required packages
# Ensure these are installed first: install.packages(c("ggplot2", "dplyr"))
# Bioconductor packages: BiocManager::install(c("phyloseq", "vegan"))
library(phyloseq)
library(vegan)
library(ggplot2)
library(dplyr)

# Set a random seed for reproducibility of any stochastic processes
set.seed(12345)

# Define file paths
# ASSUMPTION: Your R session's working directory is the project root.
input_phyloseq_path <- "data/probiotic_study.rds"
output_plot_path <- "results/pcoa_bray_curtis_plot.pdf"

# Create results directory if it doesn't exist
if (!dir.exists("results")) {
  dir.create("results")
}


# --- Section 1: Data Loading and Quality Control ---

# Load the phyloseq object from an RDS file
if (!file.exists(input_phyloseq_path)) {
  stop("Input file not found at: ", input_phyloseq_path)
}
ps_object <- readRDS(input_phyloseq_path)

# Basic check to ensure it's a phyloseq object
if (!inherits(ps_object, "phyloseq")) {
  stop("Loaded object is not a phyloseq object.")
}

# Print a summary of the data
cat("--- Phyloseq Object Summary ---\n")
print(ps_object)
cat("\n")

# Extract sample data for validation
sample_df <- as(sample_data(ps_object), "data.frame")

# Check for missing values in key metadata columns
key_vars <- c("treatment", "timepoint", "subject_id")
if (any(sapply(sample_df[, key_vars], function(x) any(is.na(x))))) {
  warning("Missing values detected in key metadata columns (treatment, timepoint, or subject_id). This may cause errors.")
} else {
  cat("--- QC Check: No missing values in key metadata columns. ---\n\n")
}

# ASSUMPTION: The OTU table in the phyloseq object is already filtered
# and normalized (e.g., rarefied or transformed to relative abundance).
# Bray-Curtis is sensitive to library size differences. If not normalized,
# consider transforming the data first.
# Example transformation to relative abundance (if needed):
# ps_object_rel <- transform_sample_counts(ps_object, function(x) x / sum(x))


# --- Section 2: Beta Diversity Calculation ---

# Calculate Bray-Curtis distance matrix
# This measures the dissimilarity between microbial communities
bray_dist <- phyloseq::distance(ps_object, method = "bray")


# --- Section 3: Ordination (PCoA) ---

# Perform Principal Coordinates Analysis (PCoA) on the distance matrix
pcoa_ord <- phyloseq::ordinate(ps_object, method = "PCoA", distance = bray_dist)

# Extract and print the variance explained by the first few axes
eigen_vals <- pcoa_ord$values$Eigenvalues
variance_explained <- eigen_vals / sum(eigen_vals)
cat("--- PCoA Variance Explained ---\n")
cat(sprintf("Axis 1: %.2f%%\n", variance_explained[1] * 100))
cat(sprintf("Axis 2: %.2f%%\n", variance_explained[2] * 100))
cat(sprintf("Axis 3: %.2f%%\n", variance_explained[3] * 100))
cat("\n")


# --- Section 4: Visualization ---

# Create the PCoA plot using ggplot2
# We will color by 'treatment' and shape by 'timepoint'
pcoa_plot <- plot_ordination(
  physeq = ps_object,
  ordination = pcoa_ord,
  color = "treatment",
  shape = "timepoint"
) +
  geom_point(size = 4, alpha = 0.75) + # Set point size and transparency
  labs(
    title = "PCoA of Microbial Communities",
    subtitle = "Bray-Curtis Dissimilarity",
    color = "Treatment Group",
    shape = "Timepoint",
    x = sprintf("PCoA Axis 1 (%.2f%%)", variance_explained[1] * 100),
    y = sprintf("PCoA Axis 2 (%.2f%%)", variance_explained[2] * 100)
  ) +
  theme_bw() + # Use a clean black and white theme
  theme(
    plot.title = element_text(hjust = 0.5, face = "bold"),
    plot.subtitle = element_text(hjust = 0.5),
    legend.position = "right"
  ) +
  # Optional: Add ellipses to show group clustering
  stat_ellipse(type = "t", aes(group = treatment), linetype = 2)

# Display the plot
print(pcoa_plot)

# Save the plot to a PDF file for publication quality
ggsave(
  filename = output_plot_path,
  plot = pcoa_plot,
  width = 8,
  height = 6,
  units = "in"
)

cat(paste("--- Plot saved to:", output_plot_path, "---\n\n"))


# --- Section 5: Statistical Analysis (PERMANOVA) ---

# Permutational Multivariate Analysis of Variance (PERMANOVA)
# We test the effect of treatment, timepoint, and their interaction.
# To account for repeated measures from the same subject, we use the 'strata'
# argument or, more flexibly, define permutation blocks.

# Get the sample data data.frame
sample_df <- data.frame(sample_data(ps_object))

# Define the permutation design to block by subject_id
# This ensures that samples are only shuffled *within* each subject,
# respecting the repeated measures design.
permutations <- how(
  nperm = 999,
  within = list(type = "free"), # Free permutation within each subject
  plots = Plots(strata = as.factor(sample_df$subject_id), type = "none")
)

# Run the PERMANOVA using adonis2 from the vegan package
# The formula `bray_dist ~ treatment * timepoint` tests for the main effects
# of treatment, timepoint, and their interaction.
permanova_results <- adonis2(
  bray_dist ~ treatment * timepoint,
  data = sample_df,
  permutations = permutations
)

# Print the results to the console
cat("--- PERMANOVA Results (Blocked by Subject ID) ---\n")
print(permanova_results)
cat("\n")

# Interpretation guide:
# - Look at the Pr(>F) column (p-value).
# - A significant p-value for 'treatment' suggests an overall effect of the probiotic.
# - A significant p-value for 'timepoint' suggests community changes over time.
# - A significant p-value for 'treatment:timepoint' (the interaction) is key:
#   it suggests that the effect of the treatment differs across timepoints
#   (e.g., the probiotic group diverges from placebo over time).


# --- Section 6: Script Completion ---

cat("Analysis complete.\n")

################################################################################
#                                END OF SCRIPT                                 #
################################################################################
