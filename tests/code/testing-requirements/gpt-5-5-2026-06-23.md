```python
import numpy as np

def clr_transform_safe(counts, pseudocount=0.5):
    """Apply centered log-ratio transformation to nonnegative count data."""
    counts = np.asarray(counts)

    if counts.ndim != 2:
        raise ValueError("counts must be a 2D array with samples as rows and features as columns.")

    if counts.shape[1] == 0:
        raise ValueError("counts must contain at least one feature/column.")

    if np.any(~np.isfinite(counts)):
        raise ValueError("counts must not contain nan or inf values.")

    if np.any(counts < 0):
        raise ValueError("counts must be nonnegative.")

    if pseudocount <= 0:
        raise ValueError("pseudocount must be positive.")

    counts_pseudo = counts + pseudocount
    log_counts = np.log(counts_pseudo)

    clr = log_counts - log_counts.mean(axis=1, keepdims=True)

    return clr
```

This version is algebraically equivalent to your current implementation, but slightly simpler and avoids explicitly calculating the geometric mean.
