```python
# your_clr_module.py

import pandas as pd
import numpy as np
from scipy.stats import gmean

def clr_transform(data: pd.DataFrame, pseudocount=1e-6) -> pd.DataFrame:
    """
    Performs Centered Log-Ratio (CLR) transformation on count data.

    Args:
        data (pd.DataFrame): A DataFrame of counts (samples x features).
        pseudocount (float): A small value to add to handle zeros.

    Returns:
        pd.DataFrame: The CLR-transformed data.
    """
    if (data == 0).any().any():
        print(f"Zeros found. Adding a pseudocount of {pseudocount}.")
        data = data + pseudocount

    # Calculate the geometric mean for each row (sample)
    # Note: Using scipy.stats.gmean is a common way to do this.
    geometric_mean = gmean(data, axis=1)

    # Perform the CLR transformation
    # clr_x = log(x / gmean(x_row))
    # which is equivalent to log(x) - log(gmean(x_row))
    # We operate on the whole dataframe for efficiency.
    log_data = np.log(data)
    log_gmean = np.log(geometric_mean)

    # Subtract the log of the geometric mean from each element in the row
    clr_data = log_data.subtract(log_gmean, axis=0)

    return clr_data
```
