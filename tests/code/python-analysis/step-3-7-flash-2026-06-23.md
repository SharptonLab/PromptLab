"""
Longitudinal Infant Gut Microbiome Diversity Trajectory Analysis
This script loads longitudinal microbiome diversity data, calculates monthly mean Shannon diversity
with 95% confidence intervals, compares diversity trajectories between delivery modes using a linear
mixed model, and generates publication-quality visualizations of the results.

Input: CSV file with columns: sample_id, subject_id, age_days, shannon, observed_features,
       delivery_mode (vaginal/cesarean), feeding_mode (breastfed/formula)
Output: Summary statistics CSV, linear mixed model results (text and CSV), publication-quality
        trajectory plot (PNG and SVG)
"""

# ------------------------------
# Standard Library Imports
# ------------------------------
import argparse
from pathlib import Path
from typing import Tuple, Any
import warnings

# ------------------------------
# Third-Party Imports
# ------------------------------
import pandas as pd
import numpy as np
import matplotlib.pyplot as plt
import seaborn as sns
from scipy import stats
import statsmodels.api as sm
from statsmodels.formula.api import mixedlm


def load_and_validate_data(input_path: Path) -> pd.DataFrame:
    """
    Load and validate the input microbiome diversity CSV file.

    Parameters
    ----------
    input_path : Path
        Path to the input CSV file.

    Returns
    -------
    pd.DataFrame
        Validated dataframe with required columns, cleaned of missing key values.

    Raises
    ------
    FileNotFoundError
        If the input file does not exist.
    ValueError
        If required columns are missing, or key variables have invalid values.
    """
    # Check file existence and format
    if not input_path.exists():
        raise FileNotFoundError(f"Input file not found at {input_path}")
    if not input_path.suffix.lower() == ".csv":
        raise ValueError(f"Input file must be a CSV, got {input_path.suffix}")

    # Load data with error handling for malformed CSVs
    try:
        df = pd.read_csv(input_path)
    except Exception as e:
        raise ValueError(f"Failed to parse CSV file: {str(e)}") from e

    # Validate required columns
    required_cols = [
        "sample_id", "subject_id", "age_days", "shannon",
        "observed_features", "delivery_mode", "feeding_mode"
    ]
    missing_cols = [col for col in required_cols if col not in df.columns]
    if missing_cols:
        raise ValueError(f"Missing required columns: {', '.join(missing_cols)}")

    # Clean critical missing values (required for analysis)
    critical_cols = ["shannon", "age_days", "subject_id", "delivery_mode"]
    initial_n = len(df)
    df = df.dropna(subset=critical_cols).reset_index(drop=True)
    if len(df) < initial_n:
        warnings.warn(f"Dropped {initial_n - len(df)} rows with missing values in critical columns")

    # Validate data types
    if not np.issubdtype(df["age_days"].dtype, np.integer):
        try:
            df["age_days"] = pd.to_numeric(df["age_days"], downcast="integer")
        except:
            raise ValueError("age_days column must be numeric/integer")
    if not np.issubdtype(df["shannon"].dtype, np.number):
        try:
            df["shannon"] = pd.to_numeric(df["shannon"], errors="coerce")
            df = df.dropna(subset=["shannon"]).reset_index(drop=True)
        except:
            raise ValueError("shannon column must be numeric")

    # Validate delivery mode values
    valid_delivery = {"vaginal", "cesarean"}
    invalid_delivery = set(df["delivery_mode"].unique()) - valid_delivery
    if invalid_delivery:
        raise ValueError(
            f"Invalid delivery_mode values found: {', '.join(invalid_delivery)}. "
            f"Expected only {', '.join(valid_delivery)}"
        )

    # Ensure subject_id is string type
    df["subject_id"] = df["subject_id"].astype(str)

    return df


def bin_age_by_month(
    df: pd.DataFrame,
    age_col: str = "age_days",
    max_age_days: int = 360
) -> pd.DataFrame:
    """
    Bin age in days into 30-day (1 month) intervals aligned with the first-year study period.

    Parameters
    ----------
    df : pd.DataFrame
        Input dataframe with age_days column.
    age_col : str, optional
        Name of the age in days column, by default "age_days".
    max_age_days : int, optional
        Maximum age (in days) to include in bins, by default 360 (12 months).

    Returns
    -------
    pd.DataFrame
        Dataframe with added ordered categorical 'age_month' column indicating the age bin.
    """
    # Warn and filter out samples outside the first-year study period
    over_age = df[df[age_col] > max_age_days]
    if len(over_age) > 0:
        warnings.warn(
            f"Dropped {len(over_age)} samples with age > {max_age_days} days (outside first year study period)"
        )
    df = df[df[age_col] <= max_age_days].reset_index(drop=True)

    # Define 30-day bins: 0-30 = Month 1, 31-60 = Month 2, ..., 331-360 = Month 12
    bins = [0] + [30 * i + 1 for i in range(1, 13)] + [361]
    labels = [f"Month {i}" for i in range(1, 13)]
    df["age_month"] = pd.cut(
        df[age_col], bins=bins, labels=labels, right=True, ordered=True
    )
    # Ensure categorical order is preserved
    df["age_month"] = pd.Categorical(df["age_month"], categories=labels, ordered=True)
    df = df.dropna(subset=["age_month"]).reset_index(drop=True)

    return df


def calculate_monthly_diversity_stats(
    df: pd.DataFrame,
    group_cols: list = None
) -> pd.DataFrame:
    """
    Calculate mean Shannon diversity, sample size, standard deviation, and 95% confidence interval
    per age month and delivery mode.

    Parameters
    ----------
    df : pd.DataFrame
        Dataframe with binned age_month column and shannon, delivery_mode columns.
    group_cols : list, optional
        Additional columns to group by, by default None (groups by age_month and delivery_mode).

    Returns
    -------
    pd.DataFrame
        Summary statistics with columns: age_month, delivery_mode, mean_shannon, n_samples,
        std_shannon, sem_shannon, ci_lower, ci_upper.
    """
    if group_cols is None:
        group_cols = ["age_month", "delivery_mode"]

    # Aggregate statistics per group
    grouped = df.groupby(group_cols, observed=True)["shannon"].agg(
        mean_shannon="mean",
        n_samples="count",
        std_shannon="std"
    ).reset_index()

    # Calculate standard error and t-distribution-based 95% CI (accurate for small samples)
    grouped["sem_shannon"] = grouped["std_shannon"] / np.sqrt(grouped["n_samples"])
    grouped["t_crit"] = stats.t.ppf(0.975, df=grouped["n_samples"] - 1)
    grouped["ci_lower"] = grouped["mean_shannon"] - grouped["t_crit"] * grouped["sem_shannon"]
    grouped["ci_upper"] = grouped["mean_shannon"] + grouped["t_crit"] * grouped["sem_shannon"]
    grouped = grouped.drop(columns=["t_crit"])

    return grouped


def run_linear_mixed_model(df: pd.DataFrame) -> Tuple[Any, pd.DataFrame]:
    """
    Fit a linear mixed model to compare Shannon diversity trajectories between delivery modes.
    Model specification: shannon ~ age_days * delivery_mode + (1 | subject_id)
    Tests for main effects of age, delivery mode, and their interaction (trajectory difference).

    Parameters
    ----------
    df : pd.DataFrame
        Input dataframe with columns: shannon, age_days, delivery_mode, subject_id.

    Returns
    -------
    Tuple[Any, pd.DataFrame]
        Fitted statsmodels MixedLM results object, and a dataframe of model coefficients
        with estimates, standard errors, z-values, p-values, and 95% CI.
    """
    # Fit model with random intercept for subject (accounts for repeated measures)
    formula = "shannon ~ age_days * delivery_mode"
    model = mixedlm(formula, data=df, groups=df["subject_id"])
    results = model.fit()

    # Warn if model fails to converge
    if not results.converged:
        warnings.warn("Linear mixed model did not converge! Results may be unreliable.")

    # Extract coefficients to structured dataframe
    coef_df = pd.DataFrame({
        "term": results.params.index,
        "estimate": results.params.values,
        "std_error": results.bse.values,
        "z_value": results.tvalues.values,
        "p_value": results.pvalues.values,
        "ci_lower": results.conf_int()[0].values,
        "ci_upper": results.conf_int()[1].values
    })

    return results, coef_df


def plot_diversity_trajectories(
    stats_df: pd.DataFrame,
    output_path: Path,
    dpi: int = 300,
    figsize: Tuple[int, int] = (8, 6)
) -> None:
    """
    Generate a publication-quality line plot of mean Shannon diversity over age month,
    stratified by delivery mode, with 95% confidence intervals.

    Parameters
    ----------
    stats_df : pd.DataFrame
        Summary statistics dataframe from calculate_monthly_diversity_stats.
    output_path : Path
        Path to save the output plot (supports .png, .svg, .pdf extensions).
    dpi : int, optional
        Resolution for raster formats (PNG), by default 300.
    figsize : Tuple[int, int], optional
        Figure dimensions in inches, by default (8,6).
    """
    # Set publication-quality style
    sns.set_style("whitegrid")
    sns.set_context("paper", font_scale=1.2)

    # Create figure
    fig, ax = plt.subplots(figsize=figsize, dpi=dpi)
    palette = {"vaginal": "#1f77b4", "cesarean": "#ff7f0e"}  # Distinct colorblind-friendly palette

    # Plot trajectory and CI for each delivery mode
    for mode in stats_df["delivery_mode"].unique():
        mode_data = stats_df[stats_df["delivery_mode"] == mode].sort_values("age_month")
        ax.plot(
            mode_data["age_month"], mode_data["mean_shannon"],
            color=palette.get(mode, "gray"), linewidth=2, label=mode.capitalize()
        )
        # Shaded 95% CI band
        ax.fill_between(
            mode_data["age_month"], mode_data["ci_lower"], mode_data["ci_upper"],
            color=palette.get(mode, "gray"), alpha=0.2
        )

    # Format plot
    ax.set_xlabel("Infant Age (Month)", fontsize=12)
    ax.set_ylabel("Mean Shannon Diversity", fontsize=12)
    ax.set_title("Gut Microbiome Shannon Diversity Trajectory by Delivery Mode", fontsize=14, pad=15)
    ax.set_ylim(bottom=max(0, stats_df["ci_lower"].min() - 0.2))
    ax.legend(title="Delivery Mode", fontsize=10, title_fontsize=11)
    sns.despine(ax=ax, top=True, right=True)  # Remove top/right spines for clean look
    plt.tight_layout()

    # Save outputs
    output_path = Path(output_path)
    output_path.parent.mkdir(parents=True, exist_ok=True)
    fig.savefig(output_path, dpi=dpi, bbox_inches="tight")
    # Save vector SVG version if output is raster format
    if output_path.suffix.lower() in [".png", ".jpeg", ".jpg"]:
        svg_path = output_path.with_suffix(".svg")
        fig.savefig(svg_path, bbox_inches="tight")
        print(f"Plot saved to {output_path} and {svg_path}")
    else:
        print(f"Plot saved to {output_path}")
    plt.close(fig)


def save_outputs(
    stats_df: pd.DataFrame,
    model_results: Any,
    coef_df: pd.DataFrame,
    output_dir: Path
) -> None:
    """
    Save summary statistics, model results, and model coefficients to output files.

    Parameters
    ----------
    stats_df : pd.DataFrame
        Monthly diversity summary statistics.
    model_results : Any
        Fitted statsmodels MixedLM results object.
    coef_df : pd.DataFrame
        Model coefficients dataframe.
    output_dir : Path
        Directory to save output files.
    """
    output_dir = Path(output_dir)
    output_dir.mkdir(parents=True, exist_ok=True)

    # Save monthly summary stats
    stats_path = output_dir / "monthly_diversity_summary_stats.csv"
    stats_df.to_csv(stats_path, index=False)
    print(f"Summary statistics saved to {stats_path}")

    # Save full model summary
    model_summary_path = output_dir / "linear_mixed_model_summary.txt"
    with open(model_summary_path, "w") as f:
        f.write(str(model_results.summary()))
    print(f"Model summary saved to {model_summary_path}")

    # Save coefficient table
    coef_path = output_dir / "model_coefficients.csv"
    coef_df.to_csv(coef_path, index=False)
    print(f"Model coefficients saved to {coef_path}")


def main():
    """Main function to run the full diversity trajectory analysis pipeline."""
    # Parse command line arguments
    parser = argparse.ArgumentParser(
        description="Analyze longitudinal infant gut microbiome diversity trajectories."
    )
    parser.add_argument(
        "input_file", type=Path,
        help="Path to input CSV file with microbiome diversity data."
    )
    parser.add_argument(
        "--output_dir", type=Path, default=Path("output"),
        help="Directory to save output files (default: ./output)"
    )
    parser.add_argument(
        "--plot_path", type=Path, default=None,
        help="Custom path to save the trajectory plot (default: <output_dir>/diversity_trajectory.png)"
    )
    args = parser.parse_args()

    # Step 1: Load and validate input data
    print("Loading and validating input data...")
    df = load_and_validate_data(args.input_file)
    print(f"Loaded {len(df)} samples from {df['subject_id'].nunique()} infants.")

    # Step 2: Bin age into 1-month intervals
    print("Binning age into 1-month intervals...")
    df_binned = bin_age_by_month(df)
    expected_months = [f"Month {i}" for i in range(1, 13)]
    missing_months = [m for m in expected_months if m not in df_binned["age_month"].unique()]
    if missing_months:
        warnings.warn(f"Missing data for age bins: {', '.join(missing_months)}")

    # Step 3: Calculate monthly diversity statistics
    print("Calculating monthly diversity statistics...")
    monthly_stats = calculate_monthly_diversity_stats(df_binned)
    print("Sample counts per delivery mode:")
    print(monthly_stats.groupby("delivery_mode")["n_samples"].sum().to_string())

    # Step 4: Fit linear mixed model
    print("Fitting linear mixed model to compare delivery mode trajectories...")
    model_results, coef_df = run_linear_mixed_model(df)
    # Print key model results
    print("\n=== Linear Mixed Model Key Results ===")
    print(coef_df.to_string(index=False))
    # Interpret interaction term (tests for differing trajectories)
    interaction_term = [t for t in coef_df["term"] if t.startswith("age_days:delivery_mode")]
    if interaction_term:
        interaction_p = coef_df[coef_df["term"] == interaction_term[0]]["p_value"].values[0]
        if interaction_p < 0.05:
            print("\nResult: Significant age-delivery mode interaction (p < 0.05): diversity trajectories differ between delivery modes.")
        else:
            print("\nResult: No significant age-delivery mode interaction (p ≥ 0.05): no evidence of differing trajectories between delivery modes.")

    # Step 5: Generate trajectory plot
    print("\nGenerating trajectory plot...")
    plot_path = args.plot_path if args.plot_path else args.output_dir / "diversity_trajectory.png"
    plot_diversity_trajectories(monthly_stats, plot_path)

    # Step 6: Save all outputs
    print("\nSaving output files...")
    save_outputs(monthly_stats, model_results, coef_df, args.output_dir)
    print(f"\nAnalysis complete! All outputs saved to {args.output_dir.resolve()}")


if __name__ == "__main__":
    main()

### Assumptions Made
1. **Age Binning**: 30-day (1-month) intervals are used, aligned with the study's first-year sampling schedule (0-30 days = Month 1, 31-60 = Month 2, ..., 331-360 = Month 12). Samples outside the first year are excluded with a warning.
2. **Delivery Mode**: Delivery mode is a binary categorical variable with only `vaginal` and `cesarean` values; any other values are treated as invalid.
3. **Missing Data**: Missing values in critical columns (shannon, age_days, subject_id, delivery_mode) are missing completely at random, so listwise deletion is appropriate.
4. **Mixed Model Specification**: A random intercept for subject is used to account for repeated measures; random slopes are not included by default (can be added if there is evidence of varying age effects across infants).
5. **Confidence Intervals**: 95% CIs for monthly means are calculated with the t-distribution, which is more accurate for small sample sizes per bin than the normal approximation.
6. **Unused Variables**: `observed_features` and `feeding_mode` are validated as present in the input but not used in the default analysis (they can be added as covariates or weights if needed).
7. **Model Reference Level**: The reference level for delivery mode in the mixed model is the alphabetically first value (`cesarean` by default); the interaction term tests for differences in slope relative to this reference.
