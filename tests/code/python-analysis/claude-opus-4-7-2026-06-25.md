def assign_month_bins(df: pd.DataFrame) -> pd.DataFrame:

def bootstrap_ci(
    values: np.ndarray, n_boot: int = 2000, ci: float = 95.0, seed: int = 0
) -> tuple[float, float]:
    """Compute a bootstrap confidence interval for the mean.

    Note: With n_boot=2000 and ~hundreds of bins this is fast (<1s). For very
    large datasets consider reducing n_boot or using analytic CIs.
    """
    values = np.asarray(values, dtype=float)
    values = values[~np.isnan(values)]
    if values.size < 2:
        return (np.nan, np.nan)
    rng = np.random.default_rng(seed)
    # Vectorized bootstrap: shape (n_boot, n)
    idx = rng.integers(0, values.size, size=(n_boot, values.size))
    boot_means = values[idx].mean(axis=1)
    lo = np.percentile(boot_means, (100 - ci) / 2)
    hi = np.percentile(boot_means, 100 - (100 - ci) / 2)
    return float(lo), float(hi)
