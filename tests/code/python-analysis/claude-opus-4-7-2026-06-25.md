````
# Infant Gut Microbiome Diversity Trajectory Analysis

```python
"""
infant_diversity_trajectories.py

Analyze and visualize infant gut microbiome alpha-diversity trajectories over
the first year of life, comparing delivery modes (vaginal vs. cesarean) using
a linear mixed-effects model.

Inputs:
    CSV with columns: sample_id, subject_id, age_days, shannon,
    observed_features, delivery_mode, feeding_mode

Outputs:
    - summary_stats.csv     : mean, SD, 95% CI of Shannon diversity per
                              age-month bin and delivery mode
    - lmm_results.txt/.csv  : linear mixed model fit summary
    - diversity_trajectories.png/.pdf : publication-quality figure

Usage:
    python infant_diversity_trajectories.py --input data.csv --outdir results/
"""

# Standard library
import argparse
import logging
import sys
from pathlib import Path
from typing import Tuple

# Third-party
import numpy as np
import pandas as pd
import matplotlib.pyplot as plt
import seaborn as sns
import statsmodels.formula.api as smf
from scipy import stats

# --------------------------------------------------------------------------- #
# Configuration
# --------------------------------------------------------------------------- #
REQUIRED_COLS = [
    "sample_id",
    "subject_id",
    "age_days",
    "shannon",
    "observed_features",
    "delivery_mode",
    "feeding_mode",
]
VALID_DELIVERY = {"vaginal", "cesarean"}
VALID_FEEDING = {"breastfed", "formula"}
AGE_BIN_EDGES = np.arange(0, 391, 30)  # 0-30, 31-60, ..., up to ~13 months
AGE_BIN_LABELS = [f"{lo+1}-{lo+30}" for lo in AGE_BIN_EDGES[:-1]]

logging.basicConfig(
    format="%(asctime)s [%(levelname)s] %(message)s", level=logging.INFO
)
logger = logging.getLogger(__name__)


# --------------------------------------------------------------------------- #
# Data loading & validation
# --------------------------------------------------------------------------- #
def load_and_validate(path: Path) -> pd.DataFrame:
    """Load CSV and validate structure.

    Args:
        path: Path to input CSV.

    Returns:
        Cleaned DataFrame with expected columns and dtypes.

    Raises:
        FileNotFoundError, ValueError
    """
    if not path.exists():
        raise FileNotFoundError(f"Input file not found: {path}")

    try:
        df = pd.read_csv(path)
    except Exception as e:
        raise ValueError(f"Could not read CSV: {e}")

    missing = [c for c in REQUIRED_COLS if c not in df.columns]
    if missing:
        raise ValueError(f"Missing required columns: {missing}")

    # Type coercion
    df["age_days"] = pd.to_numeric(df["age_days"], errors="coerce")
    df["shannon"] = pd.to_numeric(df["shannon"], errors="coerce")
    df["observed_features"] = pd.to_numeric(
        df["observed_features"], errors="coerce"
    )
    df["subject_id"] = df["subject_id"].astype(str)
    df["delivery_mode"] = df["delivery_mode"].astype(str).str.lower().str.strip()
    df["feeding_mode"] = df["feeding_mode"].astype(str).str.lower().str.strip()

    # Drop rows with missing key values
    n0 = len(df)
    df = df.dropna(subset=["age_days", "shannon", "subject_id", "delivery_mode"])
    if len(df) < n0:
        logger.warning("Dropped %d rows with missing values", n0 - len(df))

    # Validate categorical values
    bad_delivery = set(df["delivery_mode"].unique()) - VALID_DELIVERY
    if bad_delivery:
        logger.warning(
            "Unexpected delivery_mode values (dropped): %s", bad_delivery
        )
        df = df[df["delivery_mode"].isin(VALID_DELIVERY)]

    if df.empty:
        raise ValueError("No valid rows remaining after validation.")

    # Sanity check: age_days plausible for first year+ study
    if (df["age_days"] < 0).any() or (df["age_days"] > 500).any():
        logger.warning("age_days outside [0, 500] present; check data.")

    logger.info(
        "Loaded %d samples from %d subjects",
        len(df),
        df["subject_id"].nunique(),
    )
    return df


# --------------------------------------------------------------------------- #
# Summary statistics
# --------------------------------------------------------------------------- #
def bin_by_age_month(df: pd.DataFrame) -> pd.DataFrame:
    """Add an age_month_bin column (categorical) to the DataFrame."""
    df = df.copy()
    df["age_month_bin"] = pd.cut(
        df["age_days"],
        bins=AGE_BIN_EDGES,
        labels=AGE_BIN_LABELS,
        include_lowest=True,
        right=True,
    )
    return df


def compute_summary(df: pd.DataFrame) -> pd.DataFrame:
    """Compute mean, SD, N, and 95% CI of Shannon diversity per age bin × delivery.

    Uses t-distribution CI given typical n per bin ~ tens.
    """
    def _ci(x: pd.Series) -> Tuple[float, float]:
        x = x.dropna()
        n = len(x)
        if n < 2:
            return (np.nan, np.nan)
        m, se = x.mean(), stats.sem(x)
        h = se * stats.t.ppf(0.975, n - 1)
        return (m - h, m + h)

    grouped = df.groupby(["age_month_bin", "delivery_mode"], observed=True)[
        "shannon"
    ]
    summary = grouped.agg(
        n="count", mean="mean", std="std", median="median"
    ).reset_index()

    ci = grouped.apply(_ci).reset_index(name="ci")
    summary["ci_low"] = ci["ci"].apply(lambda t: t[0])
    summary["ci_high"] = ci["ci"].apply(lambda t: t[1])
    return summary


# --------------------------------------------------------------------------- #
# Linear mixed model
# --------------------------------------------------------------------------- #
def fit_lmm(df: pd.DataFrame) -> "smf.mixedlm":
    """Fit linear mixed-effects model.

    Model:
        shannon ~ age_days + delivery_mode + age_days:delivery_mode
        + random intercept per subject_id

    Rationale:
        - Fixed effect of age captures overall developmental trajectory.
        - Interaction tests whether trajectory slopes differ by delivery mode.
        - Random intercept accounts for repeated measures within infants.

    Note: For ~800 samples this fits in seconds. Random slopes could be
    added but often fail to converge with sparse per-subject data.
    """
    model_df = df.copy()
    # Reference level = cesarean so coefficient interprets vaginal effect
    model_df["delivery_mode"] = pd.Categorical(
        model_df["delivery_mode"], categories=["cesarean", "vaginal"]
    )
    # Rescale age to weeks for interpretable coefficients
    model_df["age_weeks"] = model_df["age_days"] / 7.0

    md = smf.mixedlm(
        "shannon ~ age_weeks * delivery_mode",
        data=model_df,
        groups=model_df["subject_id"],
    )
    try:
        result = md.fit(method="lbfgs", reml=True)
    except Exception as e:
        logger.error("LMM fit failed: %s", e)
        raise
    return result


def save_lmm_results(result, outdir: Path) -> None:
    """Save LMM summary text and coefficient CSV."""
    txt_path = outdir / "lmm_results.txt"
    txt_path.write_text(str(result.summary()))

    coefs = pd.DataFrame(
        {
            "coef": result.params,
            "std_err": result.bse,
            "z": result.tvalues,
            "p_value": result.pvalues,
            "ci_low": result.conf_int()[0],
            "ci_high": result.conf_int()[1],
        }
    )
    coefs.to_csv(outdir / "lmm_coefficients.csv")
    logger.info("Saved LMM results to %s", outdir)


# --------------------------------------------------------------------------- #
# Plotting
# --------------------------------------------------------------------------- #
def plot_trajectories(
    df: pd.DataFrame, summary: pd.DataFrame, outdir: Path
) -> None:
    """Create publication-quality figure of Shannon diversity vs age by delivery mode.

    Left panel: raw points + LOESS-like smoothed mean (seaborn regplot with lowess).
    Right panel: monthly binned mean ± 95% CI.
    """
    sns.set_theme(context="paper", style="whitegrid", font_scale=1.15)
    palette = {"vaginal": "#2b8cbe", "cesarean": "#e34a33"}

    fig, axes = plt.subplots(1, 2, figsize=(12, 5), sharey=True)

    # --- Panel A: continuous trajectory --------------------------------- #
    for mode, sub in df.groupby("delivery_mode"):
        axes[0].scatter(
            sub["age_days"], sub["shannon"],
            s=10, alpha=0.25, color=palette[mode], label=None,
        )
        sns.regplot(
            x="age_days", y="shannon", data=sub,
            lowess=True, scatter=False, ax=axes[0],
            line_kws={"color": palette[mode], "lw": 2.5, "label": mode},
        )
    axes[0].set_xlabel("Age (days)")
    axes[0].set_ylabel("Shannon diversity")
    axes[0].set_title("A. Individual samples + LOWESS")
    axes[0].legend(title="Delivery mode")

    # --- Panel B: binned mean ± 95% CI ---------------------------------- #
    # Use bin midpoints for x-axis
    bin_mid = {
        lbl: (AGE_BIN_EDGES[i] + AGE_BIN_EDGES[i + 1]) / 2
        for i, lbl in enumerate(AGE_BIN_LABELS)
    }
    summary = summary.copy()
    summary["x"] = summary["age_month_bin"].map(bin_mid)

    for mode, sub in summary.groupby("delivery_mode"):
        sub = sub.sort_values("x")
        axes[1].errorbar(
            sub["x"], sub["mean"],
            yerr=[sub["mean"] - sub["ci_low"], sub["ci_high"] - sub["mean"]],
            marker="o", capsize=3, lw=2, color=palette[mode], label=mode,
        )
    axes[1].set_xlabel("Age (days, bin midpoint)")
    axes[1].set_title("B. Monthly mean ± 95% CI")
    axes[1].legend(title="Delivery mode")

    fig.suptitle(
        "Infant gut microbiome Shannon diversity trajectories",
        fontsize=14, y=1.02,
    )
    fig.tight_layout()

    for ext in ("png", "pdf"):
        fig.savefig(outdir / f"diversity_trajectories.{ext}",
                    dpi=300, bbox_inches="tight")
    plt.close(fig)
    logger.info("Saved figure to %s", outdir)


# --------------------------------------------------------------------------- #
# Main
# --------------------------------------------------------------------------- #
def parse_args() -> argparse.Namespace:
    p = argparse.ArgumentParser(description=__doc__)
    p.add_argument("--input", "-i", type=Path, required=True,
                   help="Input CSV file")
    p.add_argument("--outdir", "-o", type=Path, default=Path("results"),
                   help="Output directory (created if missing)")
    return p.parse_args()


def main() -> int:
    args = parse_args()
    args.outdir.mkdir(parents=True, exist_ok=True)

    try:
        df = load_and_validate(args.input)
    except (FileNotFoundError, ValueError) as e:
        logger.error(str(e))
        return 1

    df = bin_by_age_month(df)

    summary = compute_summary(df)
    summary.to_csv(args.outdir / "summary_stats.csv", index=False)
    logger.info("Saved summary statistics")

    try:
        lmm_result = fit_lmm(df)
        save_lmm_results(lmm_result, args.outdir)
        print("\n=== Linear Mixed Model Summary ===")
        print(lmm_result.summary())
    except Exception as e:
        logger.error("Skipping LMM due to error: %s", e)

    plot_trajectories(df, summary, args.outdir)

    logger.info("Done. Outputs in %s", args.outdir.resolve())
    return 0


if __name__ == "__main__":
    sys.exit(main())

## Assumptions

1. **Delivery mode is exactly two categories** (`vaginal`, `cesarean`), lower-cased after normalization. Rows with other values are dropped with a warning.
2. **`shannon` is already computed** (e.g., from QIIME2 alpha diversity output); the script does not compute it from feature tables.
3. **Repeated measures are handled via random intercept per `subject_id`**. Random slopes were omitted because ~8 samples/infant is often too sparse for reliable slope estimation; add via `re_formula="~age_weeks"` if convergence permits.
4. **Age binning uses fixed 30-day bins** starting at day 0; the last bin extends to day 390 to accommodate any late samples. Adjust `AGE_BIN_EDGES` if your sampling schedule differs.
5. **95% CIs use the t-distribution** assuming approximate normality of Shannon means within bins (typically reasonable at n ≥ 10 per bin).
6. **`feeding_mode` is validated but not modeled** — trivial to add as a covariate: `shannon ~ age_weeks * delivery_mode + feeding_mode`.
7. **REML is used** (default) for variance component estimation; switch to ML (`reml=False`) if comparing models with different fixed effects via likelihood ratio.
````
