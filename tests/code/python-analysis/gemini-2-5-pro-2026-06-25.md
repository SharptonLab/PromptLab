#!/usr/bin/env python3
# -*- coding: utf-8 -*-

"""
Analyzes longitudinal infant gut microbiome diversity data.

This script performs the following steps:
1.  Loads microbiome diversity data from a CSV file.
2.  Validates the presence of required columns.
3.  Preprocesses the data by binning age in days into months.
4.  Calculates summary statistics (mean, count, 95% CI) for Shannon diversity
    at each time point, stratified by delivery mode.
5.  Performs a linear mixed-effects model (LMM) analysis to statistically
    compare diversity trajectories between delivery modes, accounting for

6.  Generates a publication-quality plot showing Shannon diversity over time

7.  Saves the summary statistics, LMM results, and the plot to an

"""

# 1. Imports
# Standard library imports
import argparse
import sys
from pathlib import Path
from typing import List, Tuple

# Third-party imports
import matplotlib.pyplot as plt
import numpy as np
import pandas as pd
import seaborn as sns
import statsmodels.formula.api as smf
from scipy import stats

# 2. Main Script Logic
def load_and_validate_data(filepath: Path) -> pd.DataFrame:
    """
    Loads data from a CSV file and validates its structure.

    Args:
        filepath: The path to the input CSV file.

    Returns:
        A pandas DataFrame containing the loaded data.

    Raises:
        FileNotFoundError: If the input file does not exist.
        ValueError: If the CSV is missing required columns.
    """
    if not filepath.is_file():
        raise FileNotFoundError(f"Error: Input file not found at {filepath}")

    print(f"Loading data from {filepath}...")
    df = pd.read_csv(filepath)

    expected_columns = {
        "sample_id", "subject_id", "age_days", "shannon",
        "delivery_mode"
    }
    
    missing_columns = expected_columns - set(df.columns)
    if missing_columns:
        raise ValueError(
            f"Error: Input CSV is missing required columns: {missing_columns}"
        )

    print("Data loaded and validated successfully.")
    return df


def preprocess_data(df: pd.DataFrame) -> pd.DataFrame:
    """
    Preprocesses the dataframe for analysis.

    - Converts 'age_days' to a numeric 'age_months' column.
    - Drops rows with missing essential data.
    - Ensures 'delivery_mode' is treated as a categorical variable.

    Args:
        df: The raw input DataFrame.

    Returns:
        The preprocessed DataFrame.
    """
    print("Preprocessing data...")
    df_processed = df.copy()

    # Assumption: We are binning by 30-day intervals to approximate months.
    # This is a common and reasonable simplification for this type of analysis.
    df_processed['age_months'] = df_processed['age_days'] // 30

    # Handle potential missing values in key columns
    required_subset = ['shannon', 'age_months', 'subject_id', 'delivery_mode']
    initial_rows = len(df_processed)
    df_processed.dropna(subset=required_subset, inplace=True)
    final_rows = len(df_processed)

    if initial_rows > final_rows:
        print(
            f"Warning: Dropped {initial_rows - final_rows} rows with missing "
            f"values in key columns: {required_subset}"
        )

    # Ensure categorical data has the correct type
    df_processed['delivery_mode'] = df_processed['delivery_mode'].astype('category')
    
    print("Preprocessing complete.")
    return df_processed


def calculate_summary_stats(df: pd.DataFrame) -> pd.DataFrame:
    """
    Calculates summary statistics for Shannon diversity.

    Groups data by age (in months) and delivery mode, then calculates
    mean, standard deviation, sample count, and 95% confidence interval
    for the mean of Shannon diversity.

    Args:
        df: The preprocessed DataFrame.

    Returns:
        A DataFrame with summary statistics.
    """
    print("Calculating summary statistics...")
    
    # This operation can be slow on very large datasets (>1M rows) but is
    # efficient for the specified sample size.
    summary = df.groupby(['age_months', 'delivery_mode'])['shannon'].agg(
        mean='mean',
        std='std',
        count='count'
    ).reset_index()

    # Calculate 95% Confidence Interval for the mean
    # CI = mean ± z * (std / sqrt(n))
    # For 95% CI, z-score is approximately 1.96
    z_score = stats.norm.ppf(0.975)
    summary['ci_margin'] = z_score * (summary['std'] / np.sqrt(summary['count']))
    summary['ci_lower'] = summary['mean'] - summary['ci_margin']
    summary['ci_upper'] = summary['mean'] + summary['ci_margin']

    print("Summary statistics calculated.")
    return summary


def run_lmm_analysis(df: pd.DataFrame) -> Tuple[str, pd.DataFrame]:
    """
    Performs a linear mixed-effects model (LMM) analysis.

    The model tests if Shannon diversity trajectories differ by delivery mode.
    Formula: shannon ~ age_months * delivery_mode + (1 | subject_id)
    - Fixed effects: age_months, delivery_mode, and their interaction.
    - Random effect: A random intercept for each subject_id to account for
      non-independence of samples from the same infant.

    Args:
        df: The preprocessed DataFrame.

    Returns:
        A tuple containing:
        - The full model summary as a string.
        - The model's fixed effects coefficients as a DataFrame.
    """
    print("Running Linear Mixed Model analysis...")
    
    # The model formula:
    # 'shannon ~ age_months * delivery_mode' expands to:
    # 'shannon ~ age_months + delivery_mode + age_months:delivery_mode'
    # This tests for differences in intercept (baseline diversity) and
    # slope (change over time) between delivery modes.
    model_formula = "shannon ~ age_months * delivery_mode"
    
    # The `groups` argument specifies the random effect structure.
    # Here, we model a random intercept for each subject.
    model = smf.mixedlm(
        formula=model_formula,
        data=df,
        groups=df["subject_id"]
    )
    
    results = model.fit()
    
    print("LMM analysis complete.")
    
    # Extract results for saving
    summary_text = results.summary().as_text()
    coefficients = pd.read_html(results.summary().tables[1].as_html(), header=0, index_col=0)[0]
    
    return summary_text, coefficients


def plot_diversity_trajectory(
    df: pd.DataFrame,
    output_path: Path
) -> None:
    """
    Creates and saves a publication-quality plot of diversity trajectories.

    Uses seaborn.lineplot to show mean Shannon diversity over time with 95% CI,
    stratified by delivery mode.

    Args:
        df: The preprocessed DataFrame.
        output_path: The path to save the output PNG figure.
    """
    print(f"Generating plot and saving to {output_path}...")
    
    plt.style.use('seaborn-v0_8-whitegrid')
    fig, ax = plt.subplots(figsize=(8, 6))

    # seaborn's lineplot automatically calculates and plots the mean and 95% CI
    sns.lineplot(
        data=df,
        x='age_months',
        y='shannon',
        hue='delivery_mode',
        ci=95,  # Use bootstrapping for CI, can be 'sd' for standard deviation
        ax=ax,
        marker='o',
        linestyle='-'
    )

    # --- Customize for publication quality ---
    ax.set_title('Infant Gut Microbiome Diversity Trajectory', fontsize=16, fontweight='bold')
    ax.set_xlabel('Age (Months)', fontsize=12)
    ax.set_ylabel('Shannon Diversity Index', fontsize=12)
    
    # Set x-axis ticks to be integers representing months
    max_month = df['age_months'].max()
    ax.set_xticks(np.arange(0, max_month + 1, 2))
    ax.tick_params(axis='both', which='major', labelsize=10)
    
    # Improve legend
    handles, labels = ax.get_legend_handles_labels()
    ax.legend(handles=handles, labels=labels, title='Delivery Mode', fontsize=10)
    
    # Remove top and right spines for a cleaner look
    sns.despine(ax=ax)
    
    plt.tight_layout()
    
    # Save the figure with high resolution
    plt.savefig(output_path, dpi=300, bbox_inches='tight')
    plt.close(fig)
    print("Plot generated successfully.")


def main():
    """Main function to orchestrate the analysis pipeline."""
    parser = argparse.ArgumentParser(

if __name__ == "__main__":

3.  **Categorical Variables**: The `delivery_mode` column contains a limited number of distinct, correctly spelled categories (e.g., 'vaginal', 'cesarean'). The script will treat them as is.
4.  **Missing Data**: The script assumes that rows with missing values in `shannon`, `age_days`, `subject_id`, or `delivery_mode` are not informative and can be safely dropped. It prints a warning if any rows are dropped.
5.  **Statistical Model**: A linear mixed-effects model with a random intercept for `subject_id` is an appropriate statistical approach for this longitudinal data structure. This model assumes that while each infant has a different baseline diversity, the rate of change over time (slope) is constant across all infants within a delivery group.
