Calculate and plot longitudinal Shannon diversity trajectories by delivery mode.

This script:
1. Loads an infant gut microbiome diversity metadata table from CSV.
2. Validates required columns and basic data types.
3. Bins infant age in days into approximate monthly bins:
   0-30, 31-60, 61-90, etc.
4. Calculates summary statistics for Shannon diversity by month bin and
   delivery mode.
5. Fits a linear mixed-effects model comparing Shannon diversity trajectories

   - Linear mixed model results as TXT

def clean_and_validate_data(df: pd.DataFrame, max_age_days: int) -> pd.DataFrame:
    """
    Clean and validate key variables.

    This function:
    - Ensures key columns have usable data types.
    - Drops rows with missing required analysis variables.
    - Restricts analysis to samples from age 0 through `max_age_days`.
    - Standardizes delivery mode labels to lowercase.
    - Checks delivery mode values.

    Parameters
    ----------
    df
        Raw input dataframe.
    max_age_days
        Maximum infant age in days to include.

    Returns
    -------
    pandas.DataFrame
        Cleaned dataframe suitable for summary statistics and modeling.

    Raises
    ------
    ValueError
        If no valid rows remain after cleaning, or if key values are invalid.
    """
    cleaned = df.copy()

    # Standardize string columns.
    for col in ["sample_id", "subject_id", "delivery_mode", "feeding_mode"]:
        cleaned[col] = cleaned[col].astype(str).str.strip()

    cleaned["delivery_mode"] = cleaned["delivery_mode"].str.lower()
    cleaned["feeding_mode"] = cleaned["feeding_mode"].str.lower()

    # Convert numeric columns. Invalid values become NaN and are removed below.
    cleaned["age_days"] = pd.to_numeric(cleaned["age_days"], errors="coerce")
    cleaned["shannon"] = pd.to_numeric(cleaned["shannon"], errors="coerce")
    cleaned["observed_features"] = pd.to_numeric(
        cleaned["observed_features"], errors="coerce"
    )

    analysis_columns = ["sample_id", "subject_id", "age_days", "shannon", "delivery_mode"]
    n_before = len(cleaned)
    cleaned = cleaned.dropna(subset=analysis_columns).copy()
    n_dropped_missing = n_before - len(cleaned)

    if n_dropped_missing > 0:
        print(
            f"Warning: Dropped {n_dropped_missing} rows with missing or invalid "
            "sample_id, subject_id, age_days, shannon, or delivery_mode.",
            file=sys.stderr,
        )

    # Ages should be biologically plausible for this first-year infant study.
    n_before_age_filter = len(cleaned)
    cleaned = cleaned[
        (cleaned["age_days"] >= 0) & (cleaned["age_days"] <= max_age_days)
    ].copy()
    n_dropped_age = n_before_age_filter - len(cleaned)

    if n_dropped_age > 0:
        print(
            f"Warning: Dropped {n_dropped_age} rows outside age range "
            f"0-{max_age_days} days.",
            file=sys.stderr,
        )

    invalid_delivery_modes = (
        set(cleaned["delivery_mode"].dropna().unique()) - EXPECTED_DELIVERY_MODES
    )
    if invalid_delivery_modes:
        raise ValueError(
            "Unexpected delivery_mode values found: "
            f"{sorted(invalid_delivery_modes)}. "
            f"Expected values: {sorted(EXPECTED_DELIVERY_MODES)}"
        )

    if cleaned.empty:
        raise ValueError("No valid rows remain after cleaning and filtering.")

    # Treat delivery mode as a categorical variable with a stable reference level.
    # Assumption: 'vaginal' is the reference category for modeling.
    cleaned["delivery_mode"] = pd.Categorical(
        cleaned["delivery_mode"],
        categories=["vaginal", "cesarean"],
        ordered=False,
    )

    # Ensure subject IDs are strings for grouping in the mixed model.
    cleaned["subject_id"] = cleaned["subject_id"].astype(str)

    # Sort for reproducibility and cleaner plotting.
    cleaned = cleaned.sort_values(["subject_id", "age_days", "sample_id"]).reset_index(
        drop=True
    )

    return cleaned


def add_month_bins(df: pd.DataFrame) -> pd.DataFrame:
    """
    Add approximate monthly age bins.

    Bins are defined as:
    - 0-30 days
    - 31-60 days
    - 61-90 days
    - etc.

    Parameters
    ----------
    df
        Cleaned dataframe containing `age_days`.

    Returns
    -------
    pandas.DataFrame
        Dataframe with additional month-bin columns.

    Notes
    -----
    This uses 30-day bins rather than calendar months. This is a common
    approximation but should be explicitly reported in methods.
    """
    binned = df.copy()

    # For age 0-30: bin index 0.
    # For age 31-60: bin index 1.
    # For age 61-90: bin index 2.
    # This implements the requested inclusive binning.
    binned["month_bin_index"] = np.where(
        binned["age_days"] <= 30,
        0,
        np.ceil((binned["age_days"] - 30) / 30).astype(int),
    )

    binned["month_start_day"] = np.where(
        binned["month_bin_index"] == 0,
        0,
        binned["month_bin_index"] * 30 + 1,
    )
    binned["month_end_day"] = (binned["month_bin_index"] + 1) * 30
    binned["month_midpoint_day"] = (
        binned["month_start_day"] + binned["month_end_day"]
    ) / 2

    binned["month_bin_label"] = (
        binned["month_start_day"].astype(int).astype(str)
        + "-"
        + binned["month_end_day"].astype(int).astype(str)
    )

    # Approximate age in months for modeling as a continuous time variable.
    binned["age_month"] = binned["age_days"] / 30.0

    return binned


def calculate_summary_statistics(df: pd.DataFrame) -> pd.DataFrame:
    """
    Calculate Shannon diversity summary statistics by month bin and delivery mode.

    Parameters
    ----------
    df
        Cleaned and binned dataframe.

    Returns
    -------
    pandas.DataFrame
        Summary table with n, mean, standard deviation, standard error,
        and 95% confidence intervals.
    """
    grouped = (
        df.groupby(
            [
                "delivery_mode",
                "month_bin_index",
                "month_bin_label",
                "month_start_day",
                "month_end_day",
                "month_midpoint_day",
            ],
            observed=True,
        )["shannon"]
        .agg(n="count", mean="mean", sd="std")
        .reset_index()
    )

    grouped["se"] = grouped["sd"] / np.sqrt(grouped["n"])

    # Use t-distribution critical values for small group sizes.
    # For n <= 1, CI cannot be estimated from the data; use NaN.
    grouped["t_crit"] = grouped["n"].apply(
        lambda n: stats.t.ppf(0.975, df=n - 1) if n > 1 else np.nan
    )
    grouped["ci95_half_width"] = grouped["t_crit"] * grouped["se"]
    grouped["ci95_lower"] = grouped["mean"] - grouped["ci95_half_width"]
    grouped["ci95_upper"] = grouped["mean"] + grouped["ci95_half_width"]

    grouped = grouped.sort_values(["delivery_mode", "month_bin_index"]).reset_index(
        drop=True
    )

    return grouped


def fit_linear_mixed_model(df: pd.DataFrame):
    """
    Fit a linear mixed-effects model for Shannon diversity trajectories.

    Model:
        shannon ~ age_month_centered * delivery_mode + (1 | subject_id)

    The interaction term tests whether the Shannon diversity trajectory over age
    differs by delivery mode. A random intercept for subject accounts for repeated
    sampling of the same infant.

    Parameters
    ----------
    df
        Cleaned and binned dataframe.

    Returns
    -------
    statsmodels.regression.mixed_linear_model.MixedLMResults
        Fitted mixed-effects model result.

    Raises
    ------
    ValueError
        If there are too few subjects or delivery-mode groups to fit the model.
    RuntimeError
        If the model fails to converge.
    """
    model_df = df[["shannon", "age_month", "delivery_mode", "subject_id"]].dropna().copy()

    n_subjects = model_df["subject_id"].nunique()
    n_delivery_modes = model_df["delivery_mode"].nunique()

    if n_subjects < 2:
        raise ValueError(
            f"At least 2 subjects are required for a mixed model. Found {n_subjects}."
        )

    if n_delivery_modes < 2:
        raise ValueError(
            "At least 2 delivery_mode groups are required to compare trajectories. "
            f"Found {n_delivery_modes}."
        )

    # Center age to make the intercept more interpretable and improve convergence.
    model_df["age_month_centered"] = (
        model_df["age_month"] - model_df["age_month"].mean()
    )

    # Random intercept model. A random slope for age may be biologically appealing,
    # but with sparse or uneven longitudinal sampling it may fail to converge.
    formula = "shannon ~ age_month_centered * C(delivery_mode)"

    try:
        mixed_model = smf.mixedlm(
            formula=formula,
            data=model_df,
            groups=model_df["subject_id"],
            re_formula="1",
        )
        result = mixed_model.fit(method="lbfgs", reml=False)
    except Exception as exc:
        raise RuntimeError(
            "Failed to fit the linear mixed-effects model. "
            "Check sample size, missing values, group balance, and whether each "
            "subject has repeated measurements."
        ) from exc

    if not result.converged:
        raise RuntimeError(
            "Linear mixed-effects model did not converge. "
            "Consider simplifying the model, checking outliers, or verifying "
            "that delivery modes and ages are sufficiently represented."
        )

    return result


def write_model_results(
    result,
    output_path: Path,
    n_samples: int,
    n_subjects: int,
    n_delivery_modes: int,
) -> None:
    """
    Write mixed model results to a text file.

    Parameters
    ----------
    result
        Fitted statsmodels mixed model result.
    output_path
        Path to output TXT file.
    n_samples
        Number of samples used.
    n_subjects
        Number of unique subjects used.
    n_delivery_modes
        Number of delivery modes represented.
    """
    with output_path.open("w", encoding="utf-8") as handle:
        handle.write("Linear mixed-effects model results\n")
        handle.write("==================================\n\n")
        handle.write(
            "Model formula:\n"
            "    shannon ~ age_month_centered * C(delivery_mode) "
            "+ (1 | subject_id)\n\n"
        )
        handle.write("Interpretation notes:\n")
        handle.write(
            "- The age_month_centered term estimates the Shannon diversity "
            "trajectory over time for the reference delivery mode.\n"
        )
        handle.write(
            "- The C(delivery_mode)[T.cesarean] term estimates the average "
            "difference between cesarean and vaginal delivery at the centered age.\n"
        )
        handle.write(
            "- The interaction term age_month_centered:C(delivery_mode)[T.cesarean] "
            "tests whether the slope over age differs between delivery modes.\n"
        )
        handle.write(
            "- subject_id is modeled as a random intercept to account for repeated "
            "measures from the same infant.\n\n"
        )
        handle.write(f"Samples used: {n_samples}\n")
        handle.write(f"Subjects used: {n_subjects}\n")
        handle.write(f"Delivery-mode groups: {n_delivery_modes}\n\n")
        handle.write(str(result.summary()))
        handle.write("\n")


def plot_diversity_trajectories(
    summary_df: pd.DataFrame,
    output_prefix: Path,
    dpi: int = 300,
) -> None:
    """
    Plot Shannon diversity trajectories by delivery mode.

    Parameters
    ----------
    summary_df
        Summary statistics dataframe produced by `calculate_summary_statistics`.
    output_prefix
        Output path prefix without file extension.
    dpi
        DPI for PNG output.
    """
    sns.set_theme(
        context="paper",
        style="whitegrid",
        font="sans-serif",
        rc={
            "figure.figsize": (7, 4.5),
            "axes.labelsize": 11,
            "axes.titlesize": 12,
            "xtick.labelsize": 10,
            "ytick.labelsize": 10,
            "legend.fontsize": 10,
            "pdf.fonttype": 42,
            "ps.fonttype": 42,
        },
    )

    fig, ax = plt.subplots(figsize=(7, 4.5))

    palette = {
        "vaginal": "#1b9e77",
        "cesarean": "#d95f02",
    }

    for delivery_mode, group in summary_df.groupby("delivery_mode", observed=True):
        group = group.sort_values("month_bin_index")

        x = group["month_midpoint_day"].to_numpy(dtype=float)
        y = group["mean"].to_numpy(dtype=float)

        # If n == 1 in a bin, CI is NaN. For plotting, draw CI as zero-width
        # at the mean for those bins while preserving NaNs in the output table.
        lower = group["ci95_lower"].fillna(group["mean"]).to_numpy(dtype=float)
        upper = group["ci95_upper"].fillna(group["mean"]).to_numpy(dtype=float)

        label = str(delivery_mode).capitalize()
        color = palette.get(str(delivery_mode), None)

        ax.plot(
            x,
            y,
            marker="o",
            linewidth=2,
            markersize=5,
            label=label,
            color=color,
        )
        ax.fill_between(
            x,
            lower,
            upper,
            alpha=0.2,
            color=color,
            linewidth=0,
        )

    ax.set_xlabel("Infant age, days")
    ax.set_ylabel("Shannon diversity")
    ax.set_title("Infant gut microbiome Shannon diversity trajectories")
    ax.legend(title="Delivery mode", frameon=True)
    ax.margins(x=0.02)
    sns.despine(ax=ax)

    fig.tight_layout()

    png_path = output_prefix.with_suffix(".png")
    pdf_path = output_prefix.with_suffix(".pdf")

    fig.savefig(png_path, dpi=dpi, bbox_inches="tight")
    fig.savefig(pdf_path, bbox_inches="tight")
    plt.close(fig)

def main() -> None:
    """
    Run the diversity trajectory analysis.
    """
    args = parse_args()

    validate_input_file(args.input)

## Assumptions made

- Input file is a CSV.
- Required columns are exactly:
  - `sample_id`
  - `subject_id`
  - `age_days`
  - `shannon`
  - `observed_features`
  - `delivery_mode`
  - `feeding_mode`
- `delivery_mode` contains only:
  - `vaginal`
  - `cesarean`
- Ages are analyzed from day `0` through day `365` by default.
- Monthly bins are approximate 30-day bins:
  - `0-30`
  - `31-60`
  - `61-90`
  - etc.
- The mixed model uses:
