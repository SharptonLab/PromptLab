def add_month_bin(df: pd.DataFrame) -> pd.DataFrame:
    """Add an `age_month_bin` column based on `age_days`.

    Bins of width 30 days, labeled '0-1', '1-2', ..., '11-12' months.

def compute_summary(df: pd.DataFrame) -> pd.DataFrame:
    """Compute mean, SD, 95% CI, and N of Shannon by month bin and delivery mode.

    Returns
    -------
    pd.DataFrame
        Long-format summary with columns:
        age_month_bin, delivery_mode, n, mean, sd, sem, ci95_lo, ci95_hi.
    """
    grouped = df.groupby(["age_month_bin", "delivery_mode"], observed=True)["shannon"]
    summary = grouped.agg(["count", "mean", "std"]).reset_index()
    summary = summary.rename(columns={"count": "n", "mean": "mean", "std": "sd"})

    # Standard error and 95% CI via t-distribution (small per-bin n).
    summary["sem"] = summary["sd"] / np.sqrt(summary["n"])
    tcrit = stats.t.ppf(0.975, df=np.maximum(summary["n"] - 1, 1))
    summary["ci95_lo"] = summary["mean"] - tcrit * summary["sem"]
    summary["ci95_hi"] = summary["mean"] + tcrit * summary["sem"]

    return summary

def fit_mixed_model(df: pd.DataFrame) -> Tuple[object, str]:
    """Fit a linear mixed-effects model on Shannon diversity.

    Model:
        shannon ~ age_days + delivery_mode + age_days:delivery_mode
        random intercept: subject_id

    The interaction tests whether trajectory *slopes* differ between
    delivery modes; the main effect of delivery_mode reflects baseline
    differences.

    Returns
    -------
    (result, text_summary)
        Fitted MixedLMResults and a string of the summary table.
    """
    model_df = df.dropna(subset=["shannon", "age_days", "delivery_mode", "subject_id"]).copy()

    # Center age to improve interpretation and convergence.
    model_df["age_days_c"] = model_df["age_days"] - model_df["age_days"].mean()
    # Make delivery_mode an explicit category with vaginal as reference.
    model_df["delivery_mode"] = pd.Categorical(
        model_df["delivery_mode"], categories=["vaginal", "cesarean"]
    )

    formula = "shannon ~ age_days_c * C(delivery_mode, Treatment(reference='vaginal'))"
    try:
        model = smf.mixedlm(formula, data=model_df, groups=model_df["subject_id"])
        result = model.fit(method="lbfgs", reml=True)
    except Exception as exc:
        raise RuntimeError(f"Mixed model failed to fit: {exc}") from exc

    return result, result.summary().as_text()
