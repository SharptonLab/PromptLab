"""

This script processes longitudinal infant gut microbiome data to calculate and visualize
Shannon diversity trajectories by delivery mode, using linear mixed models to compare

def load_and_validate_data(file_path: str) -> pd.DataFrame:
    """
    Load CSV data and validate required columns, data types, and value ranges.
    
    Args:
        file_path: Path to input CSV file with microbiome data.
    
    Returns:
        Cleaned pandas DataFrame with validated data.
    
    Raises:
        FileNotFoundError: If input file does not exist.
        ValueError: If required columns are missing or data fails validation checks.
    """
    # Check file exists
    if not os.path.isfile(file_path):
        raise FileNotFoundError(f"Input file not found: {file_path}")
    
    # Load CSV with error handling
    try:
        df = pd.read_csv(file_path)
    except Exception as e:
        raise ValueError(f"Failed to parse CSV file: {str(e)}") from e
    
    # Validate required columns are present
    required_cols = [
        "sample_id", "subject_id", "age_days", "shannon",
        "observed_features", "delivery_mode", "feeding_mode"
    ]
    missing_cols = [col for col in required_cols if col not in df.columns]
    if missing_cols:
        raise ValueError(f"Missing required columns: {', '.join(missing_cols)}")
    
    # Validate Shannon diversity: numeric, non-negative
    if not pd.api.types.is_numeric_dtype(df["shannon"]):
        raise ValueError("Column 'shannon' must be numeric")
    if (df["shannon"] < 0).any():
        print("Warning: Negative Shannon diversity values detected. Dropping affected rows.")
        df = df[df["shannon"] >= 0].copy()
    
    # Validate age_days: integer, non-negative, within first year (plus 40 day buffer)
    if not pd.api.types.is_integer_dtype(df["age_days"]):
        try:
            df["age_days"] = df["age_days"].astype(int)
        except ValueError:
            raise ValueError("Column 'age_days' must be integer")
    if (df["age_days"] < 0).any() or (df["age_days"] > 400).any():
        print("Warning: Age values outside 0-400 days detected. Dropping affected rows.")
        df = df[(df["age_days"] >= 0) & (df["age_days"] <= 400)].copy()
    
    # Validate delivery_mode: only allowed values
    valid_delivery_modes = {"vaginal", "cesarean"}
    invalid_modes = set(df["delivery_mode"].unique()) - valid_delivery_modes
    if invalid_modes:
        print(f"Warning: Invalid delivery_mode values {invalid_modes} detected. Dropping affected rows.")
        df = df[df["delivery_mode"].isin(valid_delivery_modes)].copy()
    
    # Validate subject_id is string type
    df["subject_id"] = df["subject_id"].astype(str)
    
    # Drop rows with missing key values
    key_cols = ["sample_id", "subject_id", "age_days", "shannon", "delivery_mode"]
    missing_rows = df[key_cols].isnull().any(axis=1)
    if missing_rows.any():
        print(f"Warning: Dropping {missing_rows.sum()} rows with missing key values.")
        df = df[~missing_rows].copy()
    
    # Final check for empty data
    if len(df) == 0:
        raise ValueError("No valid data remaining after cleaning.")
    
    return df.reset_index(drop=True)


def bin_age_to_months(df: pd.DataFrame) -> pd.DataFrame:
    """
    Bin age_days into monthly intervals (0-30, 31-60, etc.) and add midpoint age for plotting.
    Assumption: 30-day bins approximate calendar months for infant age.
    
    Args:
        df: Input DataFrame with 'age_days' column.
    
    Returns:
        DataFrame with added 'month_bin' (categorical) and 'age_midpoint_months' (numeric) columns.
    """
    # Define bin edges to match 0-30, 31-60, ... 331-360 day intervals
    # Use 0.5 offsets to correctly assign integer days to bins (e.g., 30 days goes to 0-30 bin)
    bin_edges = [0] + [i * 30 + 0.5 for i in range(1, 13)] + [400]
    bin_labels = [f"{i}-{i+1}" for i in range(12)]  # Labels: 0-1, 1-2, ... 11-12 months
    
    # Bin age into monthly intervals
    df["month_bin"] = pd.cut(
        df["age_days"],
        bins=bin_edges,
        labels=bin_labels,
        right=True,
        include_lowest=True
    )
    
    # Calculate midpoint age (in months) for each bin for plotting
    bin_midpoints = {label: (int(label.split("-")[0]) + 0.5) for label in bin_labels}
    df["age_midpoint_months"] = df["month_bin"].map(bin_midpoints)
    
    return df


def calculate_summary_stats(df: pd.DataFrame) -> pd.DataFrame:
    """
    Calculate mean Shannon diversity, standard deviation, sample size, and 95% CI
    for each month-delivery mode group.
    
    Args:
        df: Input DataFrame with 'month_bin', 'age_midpoint_months', 'shannon', and 'delivery_mode' columns.
    
    Returns:
        Summary DataFrame with group-level statistics.
    """
    # Group by month bin and delivery mode, calculate aggregate stats
    summary = df.groupby(
        ["month_bin", "delivery_mode", "age_midpoint_months"], observed=True
    )["shannon"].agg(
        mean_shannon="mean",
        std_shannon="std",
        n_samples="count"
    ).reset_index()
    
    # Calculate 95% CI for the mean (assumes normality of sample mean)
    summary["ci_margin"] = 1.96 * (summary["std_shannon"] / np.sqrt(summary["n_samples"]))
    summary["ci_lower"] = summary["mean_shannon"] - summary["ci_margin"]
    summary["ci_upper"] = summary["mean_shannon"] + summary["ci_margin"]
    
    # Remove any groups with 0 samples (should not occur with valid input)
    return summary[summary["n_samples"] > 0].copy()


def run_linear_mixed_model(df: pd.DataFrame) -> Tuple[Summary, pd.DataFrame]:
    """
    Fit a linear mixed model to compare Shannon diversity trajectories between delivery modes.
    Model specification: shannon ~ age_days * delivery_mode + (1 | subject_id)
    Assumption: No random slope for age, as individual variation in trajectory slope is minimal.
    
    Args:
        df: Input DataFrame with 'shannon', 'age_days', 'delivery_mode', and 'subject_id' columns.
    
    Returns:
        Tuple of (statsmodels results summary, coefficients DataFrame with p-values and CIs)
    """
    # Set vaginal delivery as reference group for interpretability
    df = df.copy()
    df["delivery_mode"] = pd.Categorical(df["delivery_mode"], categories=["vaginal", "cesarean"])
    
    # Fit mixed model with random intercept for subject (repeated measures)
    formula = "shannon ~ age_days * C(delivery_mode, Treatment(reference='vaginal'))"
    model = smf.mixedlm(formula=formula, data=df, groups=df["subject_id"])
    results = model.fit()
    
    # Extract coefficients to structured DataFrame
    coef_df = pd.DataFrame({
        "term": results.params.index,
        "coefficient": results.params.values,
        "std_error": results.bse.values,
        "z_value": results.tvalues.values,
        "p_value": results.pvalues.values,
        "ci_lower": results.conf_int()[0].values,
        "ci_upper": results.conf_int()[1].values
    })
    
    return results.summary(), coef_df


def plot_trajectories(
    summary_df: pd.DataFrame,
    model_results: Summary,
    output_path: str,
    dpi: int = 300
) -> None:
    """
    Generate a publication-quality plot of Shannon diversity trajectories by delivery mode.
    
    Args:
        summary_df: Summary statistics DataFrame from calculate_summary_stats.
        model_results: Linear mixed model results summary for significance annotation.
        output_path: Path to save figure (supports .png, .svg, .pdf).
        dpi: Resolution for raster image formats (default: 300 for publication).
    """
    # Set publication-ready style
    sns.set_style("whitegrid")
    sns.set_context("paper", font_scale=1.2)
    
    # Initialize figure
    fig, ax = plt.subplots(figsize=(8, 5))
    palette = {"vaginal": "#2c7bb6", "cesarean": "#d7191c"}  # Colorblind-friendly palette
    
    # Plot trajectories for each delivery mode
    for mode, group in summary_df.groupby("delivery_mode"):
        group = group.sort_values("age_midpoint_months")
        # Plot mean line with markers
        ax.plot(
            group["age_midpoint_months"],
            group["mean_shannon"],
            marker="o",
            markersize=6,
            linewidth=2,
            color=palette[mode],
            label=mode.capitalize()
        )
        # Add 95% CI error bars
        lower_err = group["mean_shannon"] - group["ci_lower"]
        upper_err = group["ci_upper"] - group["mean_shannon"]
        ax.errorbar(
            group["age_midpoint_months"],
            group["mean_shannon"],
            yerr=np.vstack([lower_err, upper_err]),
            fmt="none",
            ecolor=palette[mode],
            capsize=4,
            linewidth=1.5,
            alpha=0.7
        )
    
    # Add significance annotation if trajectory difference is significant
    interaction_term = "age_days:C(delivery_mode, Treatment(reference='vaginal'))[T.cesarean]"
    if interaction_term in model_results.tables[1].index:
        p_val = model_results.tables[1].loc[interaction_term, "P>|z|"]
        if p_val < 0.05:
            max_y = summary_df["ci_upper"].max()
            ax.text(0.5, max_y + 0.1, "*", ha="center", va="bottom", fontsize=16)
            ax.text(
                0.5, max_y + 0.05, "Trajectory difference: p < 0.05",
                ha="center", va="bottom", fontsize=10
            )
    
    # Format axes and labels
    ax.set_xlabel("Age (months)", fontsize=12)
    ax.set_ylabel("Mean Shannon Diversity Index", fontsize=12)
    ax.set_title("Infant Gut Shannon Diversity Trajectories by Delivery Mode", fontsize=14, pad=15)
    ax.set_xticks(range(0, 13))
    ax.set_xticklabels([str(i) for i in range(13)])
    ax.set_xlim(0, 12)
    ax.legend(title="Delivery Mode", frameon=True, framealpha=0.9)
    
    # Save high-resolution figure
    plt.tight_layout()
    plt.savefig(output_path, dpi=dpi, bbox_inches="tight")
    plt.close(fig)
    print(f"Publication figure saved to {output_path}")


def save_outputs(
