#!/usr/bin/env python3
"""
Calculate and plot longitudinal infant gut microbiome diversity trajectories.

This script:
1. Loads a CSV file containing infant microbiome diversity metadata.
2. Validates required columns and basic data integrity.
3. Bins age in days into approximately monthly intervals: 0-30, 31-60, etc.
4. Calculates summary statistics for Shannon diversity by age bin and delivery mode.
5. Fits a linear mixed-effects model comparing Shannon diversity trajectories by
   delivery mode, with infant subject as a random intercept.
6. Generates a publication-quality figure showing mean Shannon diversity over time
   by delivery mode with 95% confidence intervals.

Example
-------

    --input infant_microbiome.csv \
    --output-dir results \

"""

def clean_and_validate_data(
    df: pd.DataFrame,
    max_age_days: int | None = None,
) -> pd.DataFrame:
    """
    Clean and validate input data for downstream analysis.

    This function performs conservative cleaning:
    - Converts identifiers and categorical variables to strings.
    - Converts age_days and shannon to numeric.
    - Drops rows missing subject_id, age_days, shannon, or delivery_mode.
    - Drops rows with negative ages.
    - Optionally filters to a maximum age.

    Parameters
    ----------
    df
        Raw input dataframe.
    max_age_days
        Optional maximum age in days to retain.

    Returns
    -------
    pandas.DataFrame
        Cleaned dataframe.

    Raises
    ------
    ValueError
        If no valid rows remain after cleaning.
    """
    validate_columns(df, REQUIRED_COLUMNS)

    df = df.copy()

    # Keep sample and subject IDs as strings to avoid accidental numeric handling.
    df["sample_id"] = df["sample_id"].astype("string")
    df["subject_id"] = df["subject_id"].astype("string")

    # Normalize categorical labels for consistent grouping.
    df["delivery_mode"] = (
        df["delivery_mode"].astype("string").str.strip().str.lower()
    )
    df["feeding_mode"] = df["feeding_mode"].astype("string").str.strip().str.lower()

    # Coerce numeric fields. Invalid values become NaN and are removed below.
    df["age_days"] = pd.to_numeric(df["age_days"], errors="coerce")
    df["shannon"] = pd.to_numeric(df["shannon"], errors="coerce")
    df["observed_features"] = pd.to_numeric(
        df["observed_features"], errors="coerce"
    )

    n_initial = len(df)

    required_for_analysis = ["subject_id", "age_days", "shannon", "delivery_mode"]
    missing_mask = df[required_for_analysis].isna().any(axis=1)
    if missing_mask.any():
        logging.warning(
            "Dropping %d rows with missing subject_id, age_days, shannon, "
            "or delivery_mode.",
            int(missing_mask.sum()),
        )
        df = df.loc[~missing_mask].copy()

    negative_age_mask = df["age_days"] < 0
    if negative_age_mask.any():
        logging.warning(
            "Dropping %d rows with negative age_days.",
            int(negative_age_mask.sum()),
        )
        df = df.loc[~negative_age_mask].copy()

    if max_age_days is not None:
        before_filter = len(df)
        df = df.loc[df["age_days"] <= max_age_days].copy()
        logging.info(
            "Filtered to age_days <= %d: retained %d/%d rows.",
            max_age_days,
            len(df),
            before_filter,
        )

    non_expected_modes = set(df["delivery_mode"].dropna().unique()) - EXPECTED_DELIVERY_MODES
    if non_expected_modes:
        logging.warning(
            "Found delivery_mode values outside expected set %s: %s. "
            "They will be included as additional categories.",
            sorted(EXPECTED_DELIVERY_MODES),
            sorted(non_expected_modes),
        )

    duplicate_samples = df["sample_id"].duplicated().sum()
    if duplicate_samples > 0:
        logging.warning(
            "Found %d duplicated sample_id values. "
            "Proceeding, but verify that rows represent unique samples.",
            int(duplicate_samples),
        )

    if df.empty:
        raise ValueError(
            "No valid rows remain after cleaning. Check age_days, shannon, "
            "subject_id, and delivery_mode values."
        )

    logging.info("Loaded %d rows; retained %d valid rows.", n_initial, len(df))
    logging.info(
        "Data contain %d subjects and %d samples.",
        df["subject_id"].nunique(),
        len(df),
    )

    # Use a biologically interpretable ordering if both expected modes are present.
    observed_modes = list(pd.Series(df["delivery_mode"].dropna().unique()).sort_values())
    preferred_order = ["vaginal", "cesarean"]
    ordered_modes = [m for m in preferred_order if m in observed_modes] + [
        m for m in observed_modes if m not in preferred_order
    ]
    df["delivery_mode"] = pd.Categorical(
        df["delivery_mode"],
        categories=ordered_modes,
        ordered=False,
    )

    return df


def add_month_bins(df: pd.DataFrame) -> pd.DataFrame:
    """
    Add monthly age bins based on age_days.

    Bins are defined as:
    - 0-30
    - 31-60
    - 61-90
    - etc.

    Parameters
    ----------
    df
        Cleaned input dataframe containing age_days.

    Returns
    -------
    pandas.DataFrame
        Dataframe with additional age-bin columns.
    """
    df = df.copy()

    max_age = int(np.ceil(df["age_days"].max()))
    upper_bound = int(np.ceil(max_age / 30) * 30)
    if upper_bound < 30:
        upper_bound = 30

    # First bin is inclusive of both 0 and 30. Subsequent bins are 31-60, 61-90, etc.
    # Using right=True gives intervals (-inf, 30], (30, 60], ...
    bin_edges = np.arange(0, upper_bound + 30, 30)
    bin_edges = np.insert(bin_edges, 0, -0.001)

    labels = []
    starts = []
    ends = []

    for i in range(len(bin_edges) - 1):
        if i == 0:
            start = 0
        else:
            start = int(bin_edges[i]) + 1
        end = int(bin_edges[i + 1])
        labels.append(f"{start}-{end}")
        starts.append(start)
        ends.append(end)

    df["age_bin"] = pd.cut(
        df["age_days"],
        bins=bin_edges,
        labels=labels,
        include_lowest=True,
        right=True,
    )

    bin_metadata = pd.DataFrame(
        {
            "age_bin": pd.Categorical(labels, categories=labels, ordered=True),
            "age_bin_start": starts,
            "age_bin_end": ends,
            "age_bin_midpoint": [(s + e) / 2 for s, e in zip(starts, ends)],
            "age_month": np.arange(len(labels)) + 1,
        }
    )

    df = df.merge(bin_metadata, on="age_bin", how="left")

    if df["age_bin"].isna().any():
        # This should not happen for non-negative age values, but catches binning bugs.
        raise ValueError("Some rows could not be assigned to an age bin.")

    return df


def calculate_summary_statistics(
    df: pd.DataFrame,
    min_samples_per_bin: int = 2,
) -> pd.DataFrame:
    """
    Calculate Shannon diversity summary statistics by age bin and delivery mode.

    Parameters
    ----------
    df
        Input dataframe containing shannon, age_bin, subject_id, and delivery_mode.
    min_samples_per_bin
        Minimum number of samples required to calculate a confidence interval.

    Returns
    -------
    pandas.DataFrame
        Summary statistics table.
    """

    def ci95(series: pd.Series) -> float:
        """
        Calculate two-sided 95% confidence interval half-width for the mean.

        Parameters
        ----------
        series
            Numeric observations.

        Returns
        -------
        float
            95% CI half-width. NaN if insufficient observations.
        """
        values = series.dropna()
        n = len(values)
        if n < min_samples_per_bin:
            return np.nan
        sem = stats.sem(values)
        return float(stats.t.ppf(0.975, df=n - 1) * sem)

    grouped = (
        df.groupby(["age_bin", "age_bin_midpoint", "delivery_mode"], observed=True)
        .agg(
            n_samples=("shannon", "size"),
            n_subjects=("subject_id", "nunique"),
            mean_shannon=("shannon", "mean"),
            sd_shannon=("shannon", "std"),
            median_shannon=("shannon", "median"),
            min_shannon=("shannon", "min"),
            max_shannon=("shannon", "max"),
            ci95_half_width=("shannon", ci95),
        )
        .reset_index()
        .sort_values(["age_bin_midpoint", "delivery_mode"])
    )

    grouped["sem_shannon"] = grouped["sd_shannon"] / np.sqrt(grouped["n_samples"])
    grouped["ci95_lower"] = grouped["mean_shannon"] - grouped["ci95_half_width"]
    grouped["ci95_upper"] = grouped["mean_shannon"] + grouped["ci95_half_width"]

    return grouped


def fit_linear_mixed_model(df: pd.DataFrame):
    """
    Fit a linear mixed-effects model for Shannon diversity trajectories.

    Model:
        shannon ~ age_months_centered * C(delivery_mode) + (1 | subject_id)

    The model uses:
    - Fixed effects for age, delivery mode, and their interaction.
    - Random intercepts for subject_id to account for repeated sampling.

    Parameters
    ----------
    df
        Cleaned dataframe containing shannon, age_days, delivery_mode, and subject_id.

    Returns
    -------
    statsmodels.regression.mixed_linear_model.MixedLMResults
        Fitted mixed model result.

    Raises
    ------
    RuntimeError
        If model fitting fails.
    """
    model_df = df[["shannon", "age_days", "delivery_mode", "subject_id"]].dropna().copy()

    if model_df["subject_id"].nunique() < 2:
        raise RuntimeError(
            "Linear mixed model requires at least two subjects. "
            f"Found {model_df['subject_id'].nunique()}."
        )

    if model_df["delivery_mode"].nunique() < 2:
        raise RuntimeError(
            "Comparison by delivery_mode requires at least two delivery modes. "
            f"Found {model_df['delivery_mode'].nunique()}."
        )

    model_df["age_months"] = model_df["age_days"] / 30.4375
    model_df["age_months_centered"] = (
        model_df["age_months"] - model_df["age_months"].mean()
    )

    # Random-intercept model is usually stable for longitudinal microbiome data.
    # A random slope model could be considered, but may fail to converge with
    # sparse or unbalanced sampling.
    formula = "shannon ~ age_months_centered * C(delivery_mode)"

    logging.info("Fitting linear mixed model: %s + random intercept for subject_id", formula)

    try:
        model = smf.mixedlm(
            formula=formula,
            data=model_df,
            groups=model_df["subject_id"],
            re_formula="1",
        )
        result = model.fit(method="lbfgs", reml=False, maxiter=1000)
    except Exception as first_exc:
        logging.warning(
            "Initial mixed model fit failed with lbfgs: %s. Retrying with Powell optimizer.",
            first_exc,
        )
        try:
            model = smf.mixedlm(
                formula=formula,
                data=model_df,
                groups=model_df["subject_id"],
                re_formula="1",
            )
            result = model.fit(method="powell", reml=False, maxiter=2000)
        except Exception as second_exc:
            raise RuntimeError(
                "Linear mixed model fitting failed with both lbfgs and Powell "
                f"optimizers. First error: {first_exc}; second error: {second_exc}"
            ) from second_exc

    if not result.converged:
        logging.warning(
            "Mixed model did not report convergence. Interpret results cautiously."
        )

    return result


def save_model_outputs(result, output_dir: Path) -> None:
    """
    Save mixed model summary and coefficient table.

    Parameters
    ----------
    result
        Fitted statsmodels mixed model result.
    output_dir
        Directory where outputs will be saved.
    """
    summary_txt = output_dir / "linear_mixed_model_summary.txt"
    params_csv = output_dir / "linear_mixed_model_coefficients.csv"

    with summary_txt.open("w", encoding="utf-8") as handle:
        handle.write(str(result.summary()))
        handle.write("\n\n")
        handle.write("Model formula:\n")
        handle.write("shannon ~ age_months_centered * C(delivery_mode) + (1 | subject_id)\n")
        handle.write("\nNotes:\n")
        handle.write(
            "- age_months_centered = age_days / 30.4375, centered at the sample mean.\n"
        )
        handle.write(
            "- Random intercepts account for repeated samples from the same infant.\n"
        )

    coef_table = pd.DataFrame(
        {
            "term": result.params.index,
            "estimate": result.params.values,
            "standard_error": result.bse.reindex(result.params.index).values,
            "z_value": result.tvalues.reindex(result.params.index).values,
            "p_value": result.pvalues.reindex(result.params.index).values,
            "ci95_lower": result.conf_int().reindex(result.params.index)[0].values,
            "ci95_upper": result.conf_int().reindex(result.params.index)[1].values,
        }
    )
    coef_table.to_csv(params_csv, index=False)

    logging.info("Saved mixed model summary to %s", summary_txt)
    logging.info("Saved mixed model coefficients to %s", params_csv)


def plot_diversity_trajectories(
    summary_df: pd.DataFrame,
    output_path: Path,
    dpi: int = 300,
) -> None:
    """
    Plot Shannon diversity trajectories by delivery mode.

    Parameters
    ----------
    summary_df
        Summary statistics dataframe produced by calculate_summary_statistics().
    output_path
        Path to save the figure.
    dpi
        Figure resolution in dots per inch.
    """
    sns.set_theme(style="whitegrid", context="paper", font_scale=1.2)

    fig, ax = plt.subplots(figsize=(7.2, 4.8))

    delivery_modes = summary_df["delivery_mode"].dropna().unique()
    palette = sns.color_palette("Set2", n_colors=len(delivery_modes))
    color_map = dict(zip(delivery_modes, palette))

    for mode in delivery_modes:
        mode_df = summary_df.loc[summary_df["delivery_mode"] == mode].copy()
        mode_df = mode_df.sort_values("age_bin_midpoint")

        x = mode_df["age_bin_midpoint"].astype(float).to_numpy()
        y = mode_df["mean_shannon"].astype(float).to_numpy()
        lower = mode_df["ci95_lower"].astype(float).to_numpy()
        upper = mode_df["ci95_upper"].astype(float).to_numpy()

        ax.plot(
            x,
            y,
            marker="o",
            linewidth=2.2,
            markersize=5,
            label=str(mode),
            color=color_map[mode],
        )

        # Some bins may have n < 2, yielding NaN CI bounds. Matplotlib will skip them.
        ax.fill_between(
            x,
            lower,
            upper,
            alpha=0.25,
            color=color_map[mode],
            linewidth=0,
        )

    ax.set_xlabel("Age, days")
    ax.set_ylabel("Shannon diversity")
    ax.set_title("Infant gut microbiome diversity trajectories by delivery mode")

    max_x = float(summary_df["age_bin_midpoint"].max())
    x_ticks = np.arange(0, max_x + 1, 60)
    if len(x_ticks) > 0:
        ax.set_xticks(x_ticks)

    ax.legend(
        title="Delivery mode",
        frameon=True,
        loc="best",
    )

    sns.despine(ax=ax)
    fig.tight_layout()

    fig.savefig(output_path, dpi=dpi, bbox_inches="tight")
    plt.close(fig)

    logging.info("Saved diversity trajectory figure to %s", output_path)
